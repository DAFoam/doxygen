56 label DASolidDisplacementFoam::solvePrimal(
57  const Vec xvVec,
58  Vec wVec)
59 {
60  /*
61  Description:
62  Call the primal solver to get converged state variables
63 
64  Input:
65  xvVec: a vector that contains all volume mesh coordinates
66 
67  Output:
68  wVec: state variable vector
69  */
70 
71 #include "createRefsSolidDisplacement.H"
72 
73  // change the run status
74  daOptionPtr_->setOption<word>("runStatus", "solvePrimal");
75 
76  // first check if we need to change the boundary conditions based on
77  // the primalBC dict in DAOption. NOTE: this will overwrite whatever
78  // boundary conditions defined in the "0" folder
79  dictionary bcDict = daOptionPtr_->getAllOptions().subDict("primalBC");
80  if (bcDict.toc().size() != 0)
81  {
82  Info << "Setting up primal boundary conditions based on pyOptions: " << endl;
83  daFieldPtr_->setPrimalBoundaryConditions();
84  }
85 
86  Info << "\nCalculating displacement field\n"
87  << endl;
88 
89  // deform the mesh based on the xvVec
90  this->pointVec2OFMesh(xvVec);
91 
92  // check mesh quality
93  label meshOK = this->checkMesh();
94 
95  if (!meshOK)
96  {
97  return 1;
98  }
99 
100  primalMinRes_ = 1e10;
101  label printInterval = daOptionPtr_->getOption<label>("printInterval");
102  label printToScreen = 0;
103  while (this->loop(runTime)) // using simple.loop() will have seg fault in parallel
104  {
105 
106  printToScreen = this->isPrintTime(runTime, printInterval);
107 
108  if (printToScreen)
109  {
110  Info << "Iteration = " << runTime.value() << nl << endl;
111  }
112 
113  label iCorr = 0;
114  scalar initialResidual = 0;
115 
116  do
117  {
118  fvVectorMatrix DEqn(
119  fvm::d2dt2(D)
120  == fvm::laplacian(2 * mu + lambda, D, "laplacian(DD,D)")
121  + divSigmaExp);
122 
123  // get the solver performance info such as initial
124  // and final residuals
125  SolverPerformance<vector> solverU = DEqn.solve();
126  initialResidual = solverU.max().initialResidual();
127 
128  this->primalResidualControl<vector>(solverU, printToScreen, printInterval, "U");
129 
130  if (!compactNormalStress_)
131  {
132  divSigmaExp = fvc::div(DEqn.flux());
133  }
134 
135  gradD = fvc::grad(D);
136  sigmaD = mu * twoSymm(gradD) + (lambda * I) * tr(gradD);
137 
138  if (compactNormalStress_)
139  {
140  divSigmaExp = fvc::div(
141  sigmaD - (2 * mu + lambda) * gradD,
142  "div(sigmaD)");
143  }
144  else
145  {
146  divSigmaExp += fvc::div(sigmaD);
147  }
148 
149  } while (initialResidual > convergenceTolerance_ && ++iCorr < nCorr_);
150 
151  if (printToScreen)
152  {
153  this->printAllObjFuncs();
154  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
155  << " ClockTime = " << runTime.elapsedClockTime() << " s"
156  << nl << endl;
157  }
158 
159  runTime.write();
160 
161  }
162 
163 #include "calculateStressSolidDisplacement.H"
164 
165  this->calcPrimalResidualStatistics("print");
166 
167  // primal converged, assign the OpenFoam fields to the state vec wVec
168  this->ofField2StateVec(wVec);
169 
170  // write the mesh to files
171  mesh.write();
172 
173  Info << "End\n"
174  << endl;
175 
176  return this->checkResidualTol();
177 }
