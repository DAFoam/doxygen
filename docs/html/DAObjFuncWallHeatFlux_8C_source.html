86 void DAObjFuncWallHeatFlux::calcObjFunc(
87  const labelList& objFuncFaceSources,
88  const labelList& objFuncCellSources,
89  scalarList& objFuncFaceValues,
90  scalarList& objFuncCellValues,
91  scalar& objFuncValue)
92 {
93  /*
94  Description:
95  Calculate the heat flux F=k*dT/dz from the first cell. Modified based on
96  OpenFOAM/OpenFOAM-v1812/src/functionObjects/field/wallWallHeatFlux/wallWallHeatFlux.C
97 
98  Input:
99  objFuncFaceSources: List of face source (index) for this objective
100 
101  objFuncCellSources: List of cell source (index) for this objective
102 
103  Output:
104  objFuncFaceValues: the discrete value of objective for each face source (index).
105  This will be used for computing df/dw in the adjoint.
106 
107  objFuncCellValues: the discrete value of objective on each cell source (index).
108  This will be used for computing df/dw in the adjoint.
109 
110  objFuncValue: the sum of objective, reduced across all processors and scaled by "scale"
111  */
112 
113  // calculate the area of all the heat flux patches
114  if (areaSum_ < 0.0)
115  {
116  areaSum_ = 0.0;
117  forAll(objFuncFaceSources, idxI)
118  {
119  const label& objFuncFaceI = objFuncFaceSources[idxI];
120  label bFaceI = objFuncFaceI - daIndex_.nLocalInternalFaces;
121  const label patchI = daIndex_.bFacePatchI[bFaceI];
122  const label faceI = daIndex_.bFaceFaceI[bFaceI];
123  areaSum_ += mesh_.magSf().boundaryField()[patchI][faceI];
124  }
125  reduce(areaSum_, sumOp<scalar>());
126  }
127 
128  // initialize faceValues to zero
129  forAll(objFuncFaceValues, idxI)
130  {
131  objFuncFaceValues[idxI] = 0.0;
132  }
133  // initialize objFunValue
134  objFuncValue = 0.0;
135 
136  volScalarField::Boundary& wallHeatFluxBf = wallHeatFlux_.boundaryFieldRef();
137 
138 #ifdef IncompressibleFlow
139  // incompressible flow does not have he, so we do H = Cp * alphaEff * dT/dz
140  const objectRegistry& db = mesh_.thisDb();
141  const volScalarField& T = db.lookupObject<volScalarField>("T");
142  volScalarField alphaEff = daTurb_.alphaEff();
143  const volScalarField::Boundary& TBf = T.boundaryField();
144  const volScalarField::Boundary& alphaEffBf = alphaEff.boundaryField();
145  forAll(wallHeatFluxBf, patchI)
146  {
147  if (!wallHeatFluxBf[patchI].coupled())
148  {
149  wallHeatFluxBf[patchI] = Cp_ * alphaEffBf[patchI] * TBf[patchI].snGrad();
150  }
151  }
152 #endif
153 
154 #ifdef CompressibleFlow
155  // compressible flow, H = alphaEff * dHE/dz
156  volScalarField& he = thermo_.he();
157  const volScalarField::Boundary& heBf = he.boundaryField();
158  volScalarField alphaEff = daTurb_.alphaEff();
159  const volScalarField::Boundary& alphaEffBf = alphaEff.boundaryField();
160  forAll(wallHeatFluxBf, patchI)
161  {
162  if (!wallHeatFluxBf[patchI].coupled())
163  {
164  wallHeatFluxBf[patchI] = alphaEffBf[patchI] * heBf[patchI].snGrad();
165  }
166  }
167 #endif
168 
169  // calculate area weighted heat flux
170  forAll(objFuncFaceSources, idxI)
171  {
172  const label& objFuncFaceI = objFuncFaceSources[idxI];
173  label bFaceI = objFuncFaceI - daIndex_.nLocalInternalFaces;
174  const label patchI = daIndex_.bFacePatchI[bFaceI];
175  const label faceI = daIndex_.bFaceFaceI[bFaceI];
176 
177  scalar area = mesh_.magSf().boundaryField()[patchI][faceI];
178  objFuncFaceValues[idxI] = scale_ * wallHeatFluxBf[patchI][faceI] * area / areaSum_;
179 
180  objFuncValue += objFuncFaceValues[idxI];
181  }
182 
183  // need to reduce the sum of force across all processors
184  reduce(objFuncValue, sumOp<scalar>());
185 
186  return;
187 }
