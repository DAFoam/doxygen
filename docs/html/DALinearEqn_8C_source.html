27 void DALinearEqn::createMLRKSP(
28  const Mat jacMat,
29  const Mat jacPCMat,
30  KSP ksp)
31 {
32  /*
33  Description:
34  This is the main function we need to call to initialize the KSP and set
35  up parameters for solving the linear equations
36 
37  Input:
38  gmresRestart: how many Krylov spaces to keep before resetting them.
39  Usually, this is set to the gmresMaxIters
40 
41  gmresMaxIters: how many GMRES iteration to run at most
42 
43  gmresRelTol: the relative tolerance for GMRES
44 
45  gmresAbsTol: the absolute tolerance for GMRES
46 
47  globalPCIters: globa iteration for PC, usually set it to 0
48 
49  asmOverlap: ASM overlap for solving the linearEqn in parallel.
50  Usually set it to 1. Setting a higher number increases the convergence but
51  significantly increase the memory usage
52 
53  localPCIters: local iteraction for PC. usually set it to 1
54 
55  jacMatReOrdering: re-order the lhs matrix to reduce memory usage.
56  Usually we use nd, rcm, or natural (not re-ordered)
57 
58  pcFillLevel: how many leve fill-in to use for PC. This is a critical
59  parameters for convergence rate. Usually set it to 1. Setting it to a higher
60  number increase the convergence, however, the memory usage generally grows
61  exponetially. We rarely set it more than 2.
62 
63  printInfo: whether to print summary information before solving
64 
65  jacMat: the right-hand-side petsc matrix
66 
67  jacPCMat: the preconditioner matrix from which we constructor our preconditioners
68 
69  Output:
70  genksp: the set KSP object
71  */
72 
73  label gmresRestart =
74  daOption_.getSubDictOption<label>("adjEqnOption", "gmresRestart");
75  label globalPCIters =
76  daOption_.getSubDictOption<label>("adjEqnOption", "globalPCIters");
77  label asmOverlap =
78  daOption_.getSubDictOption<label>("adjEqnOption", "asmOverlap");
79  label localPCIters =
80  daOption_.getSubDictOption<label>("adjEqnOption", "localPCIters");
81  word jacMatReOrdering =
82  daOption_.getSubDictOption<word>("adjEqnOption", "jacMatReOrdering");
83  label pcFillLevel =
84  daOption_.getSubDictOption<label>("adjEqnOption", "pcFillLevel");
85  label gmresMaxIters =
86  daOption_.getSubDictOption<label>("adjEqnOption", "gmresMaxIters");
87  scalar gmresRelTol =
88  daOption_.getSubDictOption<scalar>("adjEqnOption", "gmresRelTol");
89  scalar gmresAbsTol =
90  daOption_.getSubDictOption<scalar>("adjEqnOption", "gmresAbsTol");
91  label useNonZeroInitGuess =
92  daOption_.getSubDictOption<label>("adjEqnOption", "useNonZeroInitGuess");
93  label printInfo = 1;
94 
95  PC MLRMasterPC, MLRGlobalPC;
96  PC MLRsubpc;
97  KSP MLRMasterPCKSP;
98  KSP* MLRsubksp;
99  // ASM Preconditioner variables
100  PetscInt MLRoverlap; // width of subdomain overlap
101  PetscInt MLRnlocal, MLRfirst; // number of local subblocks, first local subblock
102 
103  // Create linear solver context
104  //KSPCreate(PETSC_COMM_WORLD, &ksp);
105 
106  // Set operators. Here the matrix that defines the linear
107  // system also serves as the preconditioning matrix.
108  KSPSetOperators(ksp, jacMat, jacPCMat);
109 
110  // This code sets up the supplied kspObject in the following
111  // specific fashion.
112  //
113  // The hierarchy of the setup is:
114  // kspObject --> Supplied KSP object
115  // |
116  // --> master_PC --> Preconditioner type set to KSP
117  // |
118  // --> master_PC_KSP --> KSP type set to Richardson with 'globalPreConIts'
119  // |
120  // --> globalPC --> PC type set to 'globalPCType'
121  // | Usually Additive Schwartz and overlap is set
122  // | with 'ASMOverlap'. Use 0 to get BlockJacobi
123  // |
124  // --> subKSP --> KSP type set to Richardon with 'LocalPreConIts'
125  // |
126  // --> subPC --> PC type set to 'localPCType'.
127  // Usually ILU. 'localFillLevel' is
128  // set and 'localMatrixOrder' is used.
129  //
130  // Note that if globalPreConIts=1 then maser_PC_KSP is NOT created and master_PC=globalPC
131  // and if localPreConIts=1 then subKSP is set to preOnly.
132 
133  // First, KSPSetFromOptions MUST be called
134  KSPSetFromOptions(ksp);
135 
136  // Set GMRES
137  // Set the type of solver to GMRES
138  KSPType kspObjectType = KSPGMRES;
139 
140  KSPSetType(ksp, kspObjectType);
141  // Set the gmres restart
142  PetscInt restartGMRES = gmresRestart;
143 
144  // whether to use non-zero initial guess
145  if (useNonZeroInitGuess)
146  {
147  KSPSetInitialGuessNonzero(ksp, PETSC_TRUE);
148  }
149  else
150  {
151  KSPSetInitialGuessNonzero(ksp, PETSC_FALSE);
152  }
153 
154  KSPGMRESSetRestart(ksp, restartGMRES);
155  // Set the GMRES refinement type
156  KSPGMRESSetCGSRefinementType(ksp, KSP_GMRES_CGS_REFINE_IFNEEDED);
157 
158  // Set the preconditioner side
159  KSPSetPCSide(ksp, PC_RIGHT);
160 
161  // Set global and local PC iters
162  PetscInt globalPreConIts = globalPCIters;
163 
164  // Since there is an extraneous matMult required when using the
165  // richardson precondtiter with only 1 iteration, only use it when we need
166  // to do more than 1 iteration.
167  if (globalPreConIts > 1)
168  {
169  // Extract preconditioning context for main KSP solver: (MLRMasterPC)
170  KSPGetPC(ksp, &MLRMasterPC);
171 
172  // Set the type of MLRMasterPC to ksp. This lets us do multiple
173  // iterations of preconditioner application
174  PCSetType(MLRMasterPC, PCKSP);
175 
176  // Get the ksp context from MLRMasterPC which is the actual preconditioner:
177  PCKSPGetKSP(MLRMasterPC, &MLRMasterPCKSP);
178 
179  // MLRMasterPCKSP type will always be of type richardson. If the
180  // number of iterations is set to 1, this ksp object is transparent.
181  KSPSetType(MLRMasterPCKSP, KSPRICHARDSON);
182 
183  // Important to set the norm-type to None for efficiency.
184  KSPSetNormType(MLRMasterPCKSP, KSP_NORM_NONE);
185 
186  // Do one iteration of the outer ksp preconditioners. Note the
187  // tolerances are unsued since we have set KSP_NORM_NONE
188  KSPSetTolerances(MLRMasterPCKSP, PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT, globalPreConIts);
189 
190  // Get the 'preconditioner for MLRMasterPCKSP, called 'MLRGlobalPC'. This
191  // preconditioner is potentially run multiple times.
192  KSPGetPC(MLRMasterPCKSP, &MLRGlobalPC);
193  }
194  else
195  {
196  // Just pull out the pc-object if we are not using kspRichardson
197  KSPGetPC(ksp, &MLRGlobalPC);
198  }
199 
200  // Set the type of 'MLRGlobalPC'. This will almost always be additive schwartz
201  PCSetType(MLRGlobalPC, PCASM);
202 
203  // Set the overlap required
204  MLRoverlap = asmOverlap;
205  PCASMSetOverlap(MLRGlobalPC, MLRoverlap);
206 
207  //label KSPCalcEigen = readLabel(options.lookup("KSPCalcEigen"));
208  //if (KSPCalcEigen)
209  //{
210  // KSPSetComputeEigenvalues(*genksp, PETSC_TRUE);
211  //}
212 
213  //Setup the main ksp context before extracting the subdomains
214  KSPSetUp(ksp);
215 
216  // Extract the ksp objects for each subdomain
217  PCASMGetSubKSP(MLRGlobalPC, &MLRnlocal, &MLRfirst, &MLRsubksp);
218 
219  //Loop over the local blocks, setting various KSP options
220  //for each block.
221  PetscInt localPreConIts = localPCIters;
222  word matOrdering = jacMatReOrdering;
223  PetscInt localFillLevel = pcFillLevel;
224  for (PetscInt i = 0; i < MLRnlocal; i++)
225  {
226  // Since there is an extraneous matMult required when using the
227  // richardson precondtiter with only 1 iteration, only use it we need
228  // to do more than 1 iteration.
229  if (localPreConIts > 1)
230  {
231  // This 'subksp' object will ALSO be of type richardson so we can do
232  // multiple iterations on the sub-domains
233  KSPSetType(MLRsubksp[i], KSPRICHARDSON);
234 
235  // Set the number of iterations to do on local blocks. Tolerances are ignored.
236  KSPSetTolerances(MLRsubksp[i], PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT, localPreConIts);
237 
238  // Again, norm_type is NONE since we don't want to check error
239  KSPSetNormType(MLRsubksp[i], KSP_NORM_NONE);
240  }
241  else
242  {
243  KSPSetType(MLRsubksp[i], KSPPREONLY);
244  }
245 
246  // Extract the preconditioner for subksp object.
247  KSPGetPC(MLRsubksp[i], &MLRsubpc);
248 
249  // The subpc type will almost always be ILU
250  PCType localPCType = PCILU;
251  PCSetType(MLRsubpc, localPCType);
252 
253  // Set PC factor
254  PCFactorSetPivotInBlocks(MLRsubpc, PETSC_TRUE);
255  PCFactorSetShiftType(MLRsubpc, MAT_SHIFT_NONZERO);
256  PCFactorSetShiftAmount(MLRsubpc, PETSC_DECIDE);
257 
258  // Setup the matrix ordering for the subpc object:
259  // 'natural':'natural',
260  // 'rcm':'rcm',
261  // 'nested dissection':'nd' (default),
262  // 'one way dissection':'1wd',
263  // 'quotient minimum degree':'qmd',
264  MatOrderingType localMatrixOrdering;
265  if (matOrdering == "natural")
266  {
267  localMatrixOrdering = MATORDERINGNATURAL;
268  }
269  else if (matOrdering == "nd")
270  {
271  localMatrixOrdering = MATORDERINGND;
272  }
273  else if (matOrdering == "rcm")
274  {
275  localMatrixOrdering = MATORDERINGRCM;
276  }
277  else if (matOrdering == "1wd")
278  {
279  localMatrixOrdering = MATORDERING1WD;
280  }
281  else if (matOrdering == "qmd")
282  {
283  localMatrixOrdering = MATORDERINGQMD;
284  }
285  else
286  {
287  Info << "matOrdering not known. Using default: nested dissection" << endl;
288  localMatrixOrdering = MATORDERINGND;
289  }
290  PCFactorSetMatOrderingType(MLRsubpc, localMatrixOrdering);
291 
292  // Set the ILU parameters
293  PCFactorSetLevels(MLRsubpc, localFillLevel);
294  }
295 
296  // Set the norm to unpreconditioned
297  KSPSetNormType(ksp, KSP_NORM_UNPRECONDITIONED);
298  // Setup monitor if necessary:
299  if (printInfo)
300  {
301  KSPMonitorSet(ksp, myKSPMonitor, this, 0);
302  }
303 
304  PetscInt maxIts = gmresMaxIters;
305  PetscScalar rtol, atol;
306  assignValueCheckAD(rtol, gmresRelTol);
307  assignValueCheckAD(atol, gmresAbsTol);
308  KSPSetTolerances(ksp, rtol, atol, PETSC_DEFAULT, maxIts);
309 
310  if (printInfo)
311  {
312  Info << "Solver Type: " << kspObjectType << endl;
313  Info << "GMRES Restart: " << restartGMRES << endl;
314  Info << "ASM Overlap: " << MLRoverlap << endl;
315  Info << "Global PC Iters: " << globalPreConIts << endl;
316  Info << "Local PC Iters: " << localPreConIts << endl;
317  Info << "Mat ReOrdering: " << matOrdering << endl;
318  Info << "ILU PC Fill Level: " << localFillLevel << endl;
319  Info << "GMRES Max Iterations: " << maxIts << endl;
320  Info << "GMRES Relative Tolerance: " << rtol << endl;
321  Info << "GMRES Absolute Tolerance: " << atol << endl;
322  }

325 label DALinearEqn::solveLinearEqn(
326  const KSP ksp,
327  const Vec rhsVec,
328  Vec solVec)
329 {
330  /*
331  Description:
332  Solve a linear equation.
333 
334  Input:
335  ksp: the KSP object, obtained from calling Foam::createMLRKSP
336 
337  rhsVec: the right-hand-side petsc vector
338 
339  Output:
340  solVec: the solution vector
341 
342  Return 0 if the linear equation solution finished successfully otherwise return 1
343  */
344 
345  Info << "Solving Linear Equation... " << this->getRunTime() << " s" << endl;
346 
347  //Solve adjoint
348  // VecZeroEntries(solVec);
349 
350  // set up rGMRESHist to save the tolerance history for the GMRES solution
351  // these vars are for store the tolerance for GMRES linear solution
352  label gmresMaxIters = daOption_.getSubDictOption<label>("adjEqnOption", "gmresMaxIters");
353  PetscScalar rGMRESHist[gmresMaxIters + 1];
354  label nGMRESIters = gmresMaxIters + 1;
355  KSPSetResidualHistory(ksp, rGMRESHist, nGMRESIters, PETSC_TRUE);
356 
357  // solve KSP
358  KSPSolve(ksp, rhsVec, solVec);
359 
360  //Print convergence information
361  label its;
362  KSPGetIterationNumber(ksp, &its);
363  PetscScalar initResNorm = rGMRESHist[0];
364  PetscScalar finalResNorm = rGMRESHist[its];
365  PetscPrintf(
366  PETSC_COMM_WORLD,
367  "Main iteration %D KSP Residual norm %14.12e %d s \n",
368  its,
369  finalResNorm,
370  this->getRunTime());
371  PetscPrintf(PETSC_COMM_WORLD, "Total iterations %D\n", its);
372 
373  VecAssemblyBegin(solVec);
374  VecAssemblyEnd(solVec);
375 
376  Info << "Solving Linear Equation... Completed! "
377  << this->getRunTime() << " s" << endl;
378 
379  // now we need to check if the linear equation solution is successful
380 
381  scalar absResRatio = finalResNorm / daOption_.getSubDictOption<scalar>("adjEqnOption", "gmresAbsTol");
382  scalar relResRatio = finalResNorm / initResNorm / daOption_.getSubDictOption<scalar>("adjEqnOption", "gmresRelTol");
383  scalar resDiff = daOption_.getSubDictOption<scalar>("adjEqnOption", "gmresTolDiff");
384  if (relResRatio > resDiff && absResRatio > resDiff)
385  {
386  Info << "Residual tolerance not satisfied, solution failed!" << endl;
387  return 1;
388  }
389  else
390  {
391  Info << "Residual tolerance satisfied, solution finished!" << endl;
392  return 0;
393  }
394 
395  return 1;
396 }
