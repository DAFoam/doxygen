356 template<class classType>
357 void DASolver::primalResidualControl(
358  const SolverPerformance<classType>& solverP,
359  const label nSolverIters,
360  const label printInterval,
361  const word varName)
362 {
363  /*
364  Description:
365  Setup maximal residual control and print the residual as needed
366  */
367  if (mag(solverP.initialResidual()) < primalMinRes_)
368  {
369  primalMinRes_ = mag(solverP.initialResidual());
370  }
371  if (nSolverIters % printInterval == 0 || nSolverIters == 1)
372  {
373  Info << varName << " Initial residual: " << solverP.initialResidual() << endl
374  << varName << " Final residual: " << solverP.finalResidual() << endl;
375  }
376 }
