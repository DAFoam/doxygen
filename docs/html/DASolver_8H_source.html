696 template<class classType>
697 void DASolver::primalResidualControl(
698  const SolverPerformance<classType>& solverP,
699  const label printToScreen,
700  const label printInterval,
701  const word varName)
702 {
703  /*
704  Description:
705  Setup maximal residual control and print the residual as needed
706  */
707  if (mag(solverP.initialResidual()) < primalMinRes_)
708  {
709  primalMinRes_ = mag(solverP.initialResidual());
710  }
711  if (printToScreen)
712  {
713  Info << varName << " Initial residual: " << solverP.initialResidual() << endl
714  << varName << " Final residual: " << solverP.finalResidual() << endl;
715  }
716 }
