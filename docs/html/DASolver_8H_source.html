749 template<class classType>
750 void DASolver::primalResidualControl(
751  const SolverPerformance<classType>& solverP,
752  const label printToScreen,
753  const label printInterval,
754  const word varName)
755 {
756  /*
757  Description:
758  Setup maximal residual control and print the residual as needed
759  */
760  if (mag(solverP.initialResidual()) < primalMinRes_)
761  {
762  primalMinRes_ = mag(solverP.initialResidual());
763  }
764  if (printToScreen)
765  {
766  Info << varName << " Initial residual: " << solverP.initialResidual() << endl
767  << varName << " Final residual: " << solverP.finalResidual() << endl;
768  }
769 }
