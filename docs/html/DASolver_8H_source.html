374 template<class classType>
375 void DASolver::primalResidualControl(
376  const SolverPerformance<classType>& solverP,
377  const label printToScreen,
378  const label printInterval,
379  const word varName)
380 {
381  /*
382  Description:
383  Setup maximal residual control and print the residual as needed
384  */
385  if (mag(solverP.initialResidual()) < primalMinRes_)
386  {
387  primalMinRes_ = mag(solverP.initialResidual());
388  }
389  if (printToScreen)
390  {
391  Info << varName << " Initial residual: " << solverP.initialResidual() << endl
392  << varName << " Final residual: " << solverP.finalResidual() << endl;
393  }
394 }
