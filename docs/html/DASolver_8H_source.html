603 template<class classType>
604 void DASolver::primalResidualControl(
605  const SolverPerformance<classType>& solverP,
606  const label printToScreen,
607  const label printInterval,
608  const word varName)
609 {
610  /*
611  Description:
612  Setup maximal residual control and print the residual as needed
613  */
614  if (mag(solverP.initialResidual()) < primalMinRes_)
615  {
616  primalMinRes_ = mag(solverP.initialResidual());
617  }
618  if (printToScreen)
619  {
620  Info << varName << " Initial residual: " << solverP.initialResidual() << endl
621  << varName << " Final residual: " << solverP.finalResidual() << endl;
622  }
623 }
