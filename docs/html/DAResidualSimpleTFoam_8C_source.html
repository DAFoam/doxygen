68 void DAResidualSimpleTFoam::calcResiduals(const dictionary& options)
69 {
70  /*
71  Description:
72  This is the function to compute residuals.
73 
74  Input:
75  options.isPC: 1 means computing residuals for preconditioner matrix.
76  This essentially use the first order scheme for div(phi,U)
77 
78  p_, U_, phi_, etc: State variables in OpenFOAM
79 
80  Output:
81  URes_, pRes_, phiRes_: residual field variables
82  */
83 
84  // We dont support MRF and fvOptions so all the related lines are commented
85  // out for now
86 
87  // ******** U Residuals **********
88  // copied and modified from UEqn.H
89 
90  word divUScheme = "div(phi,U)";
91 
92  label isPC = options.getLabel("isPC");
93 
94  if (isPC)
95  {
96  divUScheme = "div(pc)";
97  }
98 
99  tmp<fvVectorMatrix> tUEqn(
100  fvm::div(phi_, U_, divUScheme)
101  + fvm::Sp(alphaPorosity_, U_)
102  + daTurb_.divDevReff(U_));
103  fvVectorMatrix& UEqn = tUEqn.ref();
104 
105  UEqn.relax();
106 
107  URes_ = (UEqn & U_) + fvc::grad(p_);
108  normalizeResiduals(URes);
109 
110  // ******** p Residuals **********
111  // copied and modified from pEqn.H
112  // NOTE manually set pRefCell and pRefValue
113  label pRefCell = 0;
114  scalar pRefValue = 0.0;
115 
116  volScalarField rAU(1.0 / UEqn.A());
117  //volVectorField HbyA(constrainHbyA(rAU*UEqn.H(), U_, p_));
118  //***************** NOTE *******************
119  // constrainHbyA has been used since OpenFOAM-v1606; however, We do NOT use the constrainHbyA
120  // function in DAFoam because we found it significantly degrades the accuracy of shape derivatives.
121  // Basically, we should not constrain any variable because it will create discontinuity.
122  // Instead, we use the old implementation used in OpenFOAM-3.0+ and before
123  volVectorField HbyA("HbyA", U_);
124  HbyA = rAU * UEqn.H();
125 
126  surfaceScalarField phiHbyA("phiHbyA", fvc::flux(HbyA));
127 
128  adjustPhi(phiHbyA, U_, p_);
129 
130  tmp<volScalarField> rAtU(rAU);
131 
132  if (simple_.consistent())
133  {
134  rAtU = 1.0 / (1.0 / rAU - UEqn.H1());
135  phiHbyA += fvc::interpolate(rAtU() - rAU) * fvc::snGrad(p_) * mesh_.magSf();
136  HbyA -= (rAU - rAtU()) * fvc::grad(p_);
137  }
138 
139  tUEqn.clear();
140 
141  fvScalarMatrix pEqn(
142  fvm::laplacian(rAtU(), p_)
143  == fvc::div(phiHbyA));
144  pEqn.setReference(pRefCell, pRefValue);
145 
146  pRes_ = pEqn & p_;
147  normalizeResiduals(pRes);
148 
149  // ******** phi Residuals **********
150  // copied and modified from pEqn.H
151  phiRes_ = phiHbyA - pEqn.flux() - phi_;
152  // need to normalize phiRes
153  normalizePhiResiduals(phiRes);
154 
155  // ******** T Residuals **************
156  volScalarField alphaEff("alphaEff", daTurb_.nu() / Pr_ + alphat_);
157 
158  fvScalarMatrix TEqn(
159  fvm::div(phi_, T_)
160  - fvm::laplacian(alphaEff, T_));
161 
162  TEqn.relax();
163 
164  TRes_ = TEqn & T_;
165  normalizeResiduals(TRes);
166 }
