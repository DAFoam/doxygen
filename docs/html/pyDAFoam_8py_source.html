854  def evalFunctions(self, funcs, evalFuncs=None, ignoreMissing=False):
855  """
856  Evaluate the desired functions given in iterable object,
857  'evalFuncs' and add them to the dictionary 'funcs'. The keys
858  in the funcs dictionary will be have an _<ap.name> appended to
859  them. Additionally, information regarding whether or not the
860  last analysis with the solvePrimal was successful is
861  included. This information is included as "funcs['fail']". If
862  the 'fail' entry already exits in the dictionary the following
863  operation is performed:
864 
865  funcs['fail'] = funcs['fail'] or <did this problem fail>
866 
867  In other words, if any one problem fails, the funcs['fail']
868  entry will be False. This information can then be used
869  directly in the pyOptSparse.
870 
871  Parameters
872  ----------
873  funcs : dict
874  Dictionary into which the functions are saved.
875 
876  evalFuncs : iterable object containing strings
877  If not None, use these functions to evaluate.
878 
879  ignoreMissing : bool
880  Flag to suppress checking for a valid function. Please use
881  this option with caution.
882 
883  Examples
884  --------
885  >>> funcs = {}
886  >>> CFDsolver()
887  >>> CFDsolver.evalFunctions(funcs, ['CD', 'CL'])
888  >>> funcs
889  >>> # Result will look like:
890  >>> # {'CD':0.501, 'CL':0.02750}
891  """
892 
893  for funcName in evalFuncs:
894  if self.primalFail:
895  if len(self.objFuncValuePrevIter) == 0:
896  raise Error("Primal solution failed for the baseline design!")
897  else:
898  # do not call self.solver.getObjFuncValue because they can be nonphysical,
899  # assign funcs based on self.objFuncValuePrevIter instead
900  funcs[funcName] = self.objFuncValuePrevIter[funcName]
901  else:
902  # call self.solver.getObjFuncValue to get the objFuncValue from
903  # the DASolver
904  objFuncValue = self.solver.getObjFuncValue(funcName.encode())
905  funcs[funcName] = objFuncValue
906  # assign the objFuncValuePrevIter
907  self.objFuncValuePrevIter[funcName] = funcs[funcName]
908 
909  if self.primalFail:
910  funcs["fail"] = True
911  else:
912  funcs["fail"] = False

1093  """
1094  This function returns a trianguled verision of the surface
1095  mesh on all processors. The intent is to use this for doing
1096  constraints in DVConstraints.
1097  Returns
1098  -------
1099  surf : list

1102  """

1105  groupName = self.allWallsGroup
1106

1108  # groupName
1109  pts = self.comm.allgather(self.getSurfaceCoordinates(groupName, **kwargs))

1111  conn = np.array(conn).flatten()

1122

1125  # Get the number of nodes on this face
1126  faceSize = faceSizes[iProc][iFace]
