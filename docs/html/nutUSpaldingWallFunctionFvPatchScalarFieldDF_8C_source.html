71 tmp<scalarField> nutUSpaldingWallFunctionFvPatchScalarFieldDF::calcUTau(
72  const scalarField& magGradU,
73  const label maxIter,
74  scalarField& err) const
75 {
76  const label patchi = patch().index();
77 
78  const turbulenceModel& turbModel = db().lookupObject<turbulenceModel>(
79  IOobject::groupName(
80  turbulenceModel::propertiesName,
81  internalField().group()));
82  const scalarField& y = turbModel.y()[patchi];
83 
84  const fvPatchVectorField& Uw = turbModel.U().boundaryField()[patchi];
85  const scalarField magUp(mag(Uw.patchInternalField() - Uw));
86 
87  const tmp<scalarField> tnuw = turbModel.nu(patchi);
88  const scalarField& nuw = tnuw();
89 
90  const scalarField& nutw = *this;
91 
92  tmp<scalarField> tuTau(new scalarField(patch().size(), Zero));
93  scalarField& uTau = tuTau.ref();
94 
95  err.setSize(uTau.size());
96  err = 0.0;
97 
98  forAll(uTau, facei)
99  {
100  scalar ut = sqrt((nutw[facei] + nuw[facei]) * magGradU[facei]);
101  // Note: for exact restart seed with laminar viscosity only:
102  //scalar ut = sqrt(nuw[facei]*magGradU[facei]);
103 
104  if (ROOTVSMALL < ut)
105  {
106  int iter = 0;
107 
108  do
109  {
110  scalar kUu = min(kappa_ * magUp[facei] / ut, 50);
111  scalar fkUu = exp(kUu) - 1 - kUu * (1 + 0.5 * kUu);
112 
113  scalar f =
114  -ut * y[facei] / nuw[facei]
115  + magUp[facei] / ut
116  + 1 / E_ * (fkUu - 1.0 / 6.0 * kUu * sqr(kUu));
117 
118  scalar df =
119  y[facei] / nuw[facei]
120  + magUp[facei] / sqr(ut)
121  + 1 / E_ * kUu * fkUu / ut;
122 
123  scalar uTauNew = ut + f / df;
124  err[facei] = mag((ut - uTauNew) / ut);
125  ut = uTauNew;
126 
127  //iterations_++;
128 
129  } while (
130  ut > ROOTVSMALL
131  && err[facei] > tolerance_
132  && ++iter < maxIter);
133 
134  uTau[facei] = max(0.0, ut);
135  }
136  }
137 
138  return tuTau;
139 }
