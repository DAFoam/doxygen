114 tmp<volScalarField> DASpalartAllmaras::Stilda(
115  const volScalarField& chi,
116  const volScalarField& fv1) const
117 {
118  volScalarField Omega(::sqrt(2.0) * mag(skew(fvc::grad(U_))));
119 
120  return (
121  max(
122  Omega

124  Cs_ * Omega));

127 tmp<volScalarField> DASpalartAllmaras::fw(
128  const volScalarField& Stilda) const
129 {
130  volScalarField r(
131  min(
132  nuTilda_
133  / (max(
134  Stilda,
135  dimensionedScalar("SMALL", Stilda.dimensions(), SMALL))
136  * sqr(kappa_ * y_)),
137  scalar(10.0)));
138  r.boundaryFieldRef() == 0.0;
139 
140  const volScalarField g(r + Cw2_ * (pow6(r) - r));
141 
142  return g * pow((1.0 + pow6(Cw3_)) / (pow6(g) + pow6(Cw3_)), 1.0 / 6.0);
143 }

157

356 void DASpalartAllmaras::correct()
357 {
358  /*
359  Descroption:
360  Solve the residual equations and update the state. This function will be called
361  by the DASolver. It is needed because we want to control the output frequency
362  of the residual convergence every 100 steps. If using the correct from turbulence
363  it will output residual every step which will be too much of information.
364  */
365 
366  // We set the flag solveTurbState_ to 1 such that in the calcResiduals function
367  // we will solve and update nuTilda
368  solveTurbState_ = 1;
369  dictionary dummyOptions;
370  this->calcResiduals(dummyOptions);
371  // after it, we reset solveTurbState_ = 0 such that calcResiduals will not
372  // update nuTilda when calling from the adjoint class, i.e., solveAdjoint from DASolver.
373  solveTurbState_ = 0;
374 }

376 void DASpalartAllmaras::calcResiduals(const dictionary& options)
377 {
378  /*
379  Descroption:
380  If solveTurbState_ == 1, this function solve and update nuTilda, and
381  is the same as calling turbulence.correct(). If solveTurbState_ == 0,
382  this function compute residuals for turbulence variables, e.g., nuTildaRes_
383 
384  Input:
385  options.isPC: 1 means computing residuals for preconditioner matrix.
386  This essentially use the first order scheme for div(phi,nuTilda)
387 
388  p_, U_, phi_, etc: State variables in OpenFOAM
389 
390  Output:
391  nuTildaRes_: If solveTurbState_ == 0, update the residual field variable
392 
393  nuTilda_: If solveTurbState_ == 1, update nuTilda
394  */
395 
396  // Copy and modify based on the "correct" function
397 
398  word divNuTildaScheme = "div(phi,nuTilda)";
399 
400  label isPC = 0;
401 
402  if (!solveTurbState_)
403  {
404  // we need to bound nuTilda before computing residuals
405  // this will avoid having NaN residuals
406  DAUtility::boundVar(allOptions_, nuTilda_);
407 
408  isPC = options.getLabel("isPC");
409 
410  if (isPC)
411  {
412  divNuTildaScheme = "div(pc)";
413  }
414  }
415 
416  //eddyViscosity<RASModelAugmented<BasicTurbulenceModel> >::correct();
417 
418  const volScalarField chi(this->chi());
419  const volScalarField fv1(this->fv1(chi));
420 
421  const volScalarField Stilda(this->Stilda(chi, fv1));
422 
423  tmp<fvScalarMatrix> nuTildaEqn(
424  fvm::ddt(phase_, rho_, nuTilda_)
425  + fvm::div(phaseRhoPhi_, nuTilda_, divNuTildaScheme)
426  - fvm::laplacian(phase_ * rho_ * DnuTildaEff(), nuTilda_)
427  - Cb2_ / sigmaNut_ * phase_ * rho_ * magSqr(fvc::grad(nuTilda_))
428  == Cb1_ * phase_ * rho_ * Stilda * nuTilda_
429  - fvm::Sp(Cw1_ * phase_ * rho_ * fw(Stilda) * nuTilda_ / sqr(y_), nuTilda_));
430 
431  nuTildaEqn.ref().relax();
432 
433  if (solveTurbState_)
434  {
435  const scalar& deltaT = mesh_.time().deltaT().value();
436  const scalar t = mesh_.time().timeOutputValue();
437  label nSolverIters = round(t / deltaT);
438 
439  // get the solver performance info such as initial
440  // and final residuals
441  SolverPerformance<scalar> solverNuTilda = solve(nuTildaEqn);
442  if (nSolverIters % printInterval_ == 0 || nSolverIters == 1)
443  {
444  Info << "nuTilda Initial residual: " << solverNuTilda.initialResidual() << endl
445  << " Final residual: " << solverNuTilda.finalResidual() << endl;
446  }
447 
448  DAUtility::boundVar(allOptions_, nuTilda_);
449  nuTilda_.correctBoundaryConditions();
450 
451  // ***************** NOTE*****************
452  // In the original SA, it is correctNut(fv1) and fv1 is not
453  // updated based on the latest nuTilda. We use correctNut which
454  // recompute fv1 with the latest nuTilda
455  this->correctNut();
456  }
457  else
458  {
459  // calculate residuals
460  nuTildaRes_ = nuTildaEqn.ref() & nuTilda_;
461  // need to normalize residuals
462  normalizeResiduals(nuTildaRes);
463  }
464 
465  return;
466 }
