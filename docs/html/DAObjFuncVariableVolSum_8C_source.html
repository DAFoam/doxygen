59 void DAObjFuncVariableVolSum::calcObjFunc(
60  const labelList& objFuncFaceSources,
61  const labelList& objFuncCellSources,
62  scalarList& objFuncFaceValues,
63  scalarList& objFuncCellValues,
64  scalar& objFuncValue)
65 {
66  /*
67  Description:
68  Calculate the obj = mesh volume * variable (whether to take a square of the variable
69  depends on isSquare)
70 
71  Input:
72  objFuncFaceSources: List of face source (index) for this objective
73 
74  objFuncCellSources: List of cell source (index) for this objective
75 
76  Output:
77  objFuncFaceValues: the discrete value of objective for each face source (index).
78  This will be used for computing df/dw in the adjoint.
79 
80  objFuncCellValues: the discrete value of objective on each cell source (index).
81  This will be used for computing df/dw in the adjoint.
82 
83  objFuncValue: the sum of objective, reduced across all processors and scaled by "scale"
84  */
85 
86  // initialize objFunValue
87  objFuncValue = 0.0;
88 
89  // initialize faceValues to zero
90  forAll(objFuncCellValues, idxI)
91  {
92  objFuncCellValues[idxI] = 0.0;
93  }
94 
95  const objectRegistry& db = mesh_.thisDb();
96 
97  if (varType_ == "scalar")
98  {
99  const volScalarField& var = db.lookupObject<volScalarField>(varName_);
100  // calculate mass
101  forAll(objFuncCellSources, idxI)
102  {
103  const label& cellI = objFuncCellSources[idxI];
104  scalar volume = mesh_.V()[cellI];
105  if (isSquare_)
106  {
107  objFuncCellValues[idxI] = scale_ * volume * var[cellI] * var[cellI];
108  }
109  else
110  {
111  objFuncCellValues[idxI] = scale_ * volume * var[cellI];
112  }
113  objFuncValue += objFuncCellValues[idxI];
114  }
115  }
116  else if (varType_ == "vector")
117  {
118  const volVectorField& var = db.lookupObject<volVectorField>(varName_);
119  // calculate mass
120  forAll(objFuncCellSources, idxI)
121  {
122  const label& cellI = objFuncCellSources[idxI];
123  scalar volume = mesh_.V()[cellI];
124  if (isSquare_)
125  {
126  objFuncCellValues[idxI] = scale_ * volume * var[cellI][component_] * var[cellI][component_];
127  }
128  else
129  {
130  objFuncCellValues[idxI] = scale_ * volume * var[cellI][component_];
131  }
132  objFuncValue += objFuncCellValues[idxI];
133  }
134  }
135  else
136  {
137  FatalErrorIn("") << "varType " << varType_ << " not supported!"
138  << "Options are: scalar or vector"
139  << abort(FatalError);
140  }
141 
142  // need to reduce the sum of force across all processors
143  reduce(objFuncValue, sumOp<scalar>());
144 
145  return;
146 }
