31 void DAFvSourceActuatorLine::calcFvSource(volVectorField& fvSource)
32 {
33  /*
34  Description:
35  Compute the actuator line source term.
36  Reference: Stokkermans et al. Validation and comparison of RANS propeller
37  modeling methods for tip-mounted applications
38  */
39 
40  const scalar& pi = Foam::constant::mathematical::pi;
41 
42  scalar t = mesh_.time().timeOutputValue();
43 
44  forAll(fvSource, idxI)
45  {
46  fvSource[idxI] = vector::zero;
47  }
48 
49  dictionary fvSourceSubDict = daOption_.getAllOptions().subDict("fvSource");
50 
51  // loop over all the cell indices for all actuator lines
52  forAll(fvSourceSubDict.toc(), idxI)
53  {
54  // name of this line model
55  word lineName = fvSourceSubDict.toc()[idxI];
56 
57  // sub dictionary with all parameters for this disk
58  dictionary lineSubDict = fvSourceSubDict.subDict(lineName);
59 
60  // now read in all parameters for this actuator line
61  // center of the actuator line
62  scalarList center1;
63  lineSubDict.readEntry<scalarList>("center", center1);
64  vector center = {center1[0], center1[1], center1[2]};
65  // thrust direction
66  scalarList direction1;
67  lineSubDict.readEntry<scalarList>("direction", direction1);
68  vector direction = {direction1[0], direction1[1], direction1[2]};
69  direction = direction / mag(direction);
70  // initial vector for the actuator line
71  scalarList initial1;
72  lineSubDict.readEntry<scalarList>("initial", initial1);
73  vector initial = {initial1[0], initial1[1], initial1[2]};
74  initial = initial / mag(initial);
75  if (fabs(direction & initial) > 1.0e-10)
76  {
77  FatalErrorIn(" ") << "direction and initial need to be orthogonal!" << abort(FatalError);
78  }
79  // rotation direction, can be either right or left
80  word rotDir = lineSubDict.getWord("rotDir");
81  // inner and outer radius of the lines
82  scalar innerRadius = lineSubDict.get<scalar>("innerRadius");
83  scalar outerRadius = lineSubDict.get<scalar>("outerRadius");
84  // rotation speed in rpm
85  scalar rpm = lineSubDict.get<scalar>("rpm");
86  scalar radPerS = rpm / 60.0 * 2.0 * pi;
87  // smooth factor which should be of grid size
88  scalar eps = lineSubDict.get<scalar>("eps");
89  // scaling factor to ensure a desired integrated thrust
90  scalar scale = lineSubDict.get<scalar>("scale");
91  // phase (rad) of the rotation positive value means rotates ahead of time for phase rad
92  scalar phase = lineSubDict.get<scalar>("phase");
93  // number of blades for this line model
94  label nBlades = lineSubDict.get<label>("nBlades");
95  scalar POD = lineSubDict.getScalar("POD");
96  scalar expM = lineSubDict.getScalar("expM");
97  scalar expN = lineSubDict.getScalar("expN");
98  // Now we need to compute normalized eps in the radial direction, i.e. epsRStar this is because
99  // we need to smooth the radial distribution of the thrust, here the radial location is
100  // normalized as rStar = (r - rInner) / (rOuter - rInner), so to make epsRStar consistent with this
101  // we need to normalize eps with the demoninator of rStar, i.e. Outer - rInner
102  scalar epsRStar = eps / (outerRadius - innerRadius);
103  scalar rStarMin = epsRStar;
104  scalar rStarMax = 1.0 - epsRStar;
105  scalar fRMin = pow(rStarMin, expM) * pow(1.0 - rStarMin, expN);
106  scalar fRMax = pow(rStarMax, expM) * pow(1.0 - rStarMax, expN);
107 
108  scalar thrustTotal = 0.0;
109  scalar torqueTotal = 0.0;
110  forAll(mesh_.C(), cellI)
111  {
112  // the cell center coordinates of this cellI
113  vector cellC = mesh_.C()[cellI];
114  // cell center to disk center vector
115  vector cellC2AVec = cellC - center;
116  // tmp tensor for calculating the axial/radial components of cellC2AVec
117  tensor cellC2AVecE(tensor::zero);
118  cellC2AVecE.xx() = cellC2AVec.x();
119  cellC2AVecE.yy() = cellC2AVec.y();
120  cellC2AVecE.zz() = cellC2AVec.z();
121  // now we need to decompose cellC2AVec into axial and radial components
122  // the axial component of cellC2AVec vector
123  vector cellC2AVecA = cellC2AVecE & direction;
124  // the radial component of cellC2AVec vector
125  vector cellC2AVecR = cellC2AVec - cellC2AVecA;
126  // now we can use the cross product to compute the tangential
127  // (circ) direction of cellI
128  vector cellC2AVecC(vector::zero);
129  if (rotDir == "left")
130  {
131  // this assumes right hand rotation of propellers
132  cellC2AVecC = cellC2AVecR ^ cellC2AVecA; // circ
133  }
134  else if (rotDir == "right")
135  {
136  // this assumes left hand rotation of propellers
137  cellC2AVecC = cellC2AVecA ^ cellC2AVecR; // circ
138  }
139  else
140  {
141  FatalErrorIn(" ") << "rotDir not valid" << abort(FatalError);
142  }
143  // the magnitude of radial component of cellC2AVecR
144  scalar cellC2AVecRLen = mag(cellC2AVecR);
145  // the magnitude of tangential component of cellC2AVecR
146  scalar cellC2AVecCLen = mag(cellC2AVecC);
147  // the magnitude of axial component of cellC2AVecA
148  scalar cellC2AVecALen = mag(cellC2AVecA);
149  // the normalized cellC2AVecC (tangential) vector
150  vector cellC2AVecCNorm = cellC2AVecC / (cellC2AVecCLen + SMALL);
151  // smooth coefficient in the axial direction
152  scalar etaAxial = exp(-sqr(cellC2AVecALen / eps));
153 
154  scalar etaTheta = 0.0;
155  for (label bb = 0; bb < nBlades; bb++)
156  {
157  scalar thetaBlade = bb * 2.0 * pi / nBlades + radPerS * t + phase;
158  if (daOption_.getOption<word>("runStatus") == "solvePrimal")
159  {
160  if (cellI == 0)
161  {
162  if (mesh_.time().timeIndex() % printIntervalUnsteady_ == 0
163  || mesh_.time().timeIndex() == 1)
164  {
165  scalar twoPi = 2.0 * pi;
166  Info << "blade " << bb << " theta: "
167 #if defined(CODI_AD_FORWARD) || defined(CODI_AD_REVERSE)
168  << fmod(thetaBlade.getValue(), twoPi.getValue()) * 180.0 / pi.getValue()
169 #else
170  << fmod(thetaBlade, twoPi) * 180.0 / pi
171 #endif
172  << " deg" << endl;
173  }
174  }
175  }
176  // compute the rotated vector of initial by thetaBlade degree
177  // We use a simplified version of Rodrigues rotation formulation
178  vector rotatedVec = vector::zero;
179  if (rotDir == "right")
180  {
181  rotatedVec = initial * cos(thetaBlade)
182  + (direction ^ initial) * sin(thetaBlade);
183  }
184  else if (rotDir == "left")
185  {
186  rotatedVec = initial * cos(thetaBlade)
187  + (initial ^ direction) * sin(thetaBlade);
188  }
189  else
190  {
191  FatalErrorIn(" ") << "rotDir not valid" << abort(FatalError);
192  }
193  // scale the rotated vector to have the same length as cellC2AVecR
194  rotatedVec *= cellC2AVecRLen;
195  // now we can compute the distance between the cellC2AVecR and the rotatedVec
196  scalar dS_Theta = mag(cellC2AVecR - rotatedVec);
197  // smooth coefficient in the theta direction
198  etaTheta += exp(-sqr(dS_Theta / eps));
199  }
200 
201  // now we can use Hoekstra's formulation to compute radial thrust distribution
202  scalar rPrime = cellC2AVecRLen / outerRadius;
203  scalar rPrimeHub = innerRadius / outerRadius;
204  // rStar is normalized radial location
205  scalar rStar = (rPrime - rPrimeHub) / (1.0 - rPrimeHub);
206 
207  scalar fAxial = 0.0;
208  scalar fCirc = 0.0;
209  scalar rPrimeMin = rStarMin * (1.0 - rPrimeHub) + rPrimeHub;
210  scalar rPrimeMax = rStarMax * (1.0 - rPrimeHub) + rPrimeHub;
211  if (rStar < rStarMin)
212  {
213  scalar dR2 = (rStar - rStarMin) * (rStar - rStarMin);
214  fAxial = fRMin * exp(-dR2 / epsRStar / epsRStar);
215  fCirc = fAxial * POD / pi / rPrimeMin;
216  }
217  else if (rStar >= rStarMin && rStar <= rStarMax)
218  {
219  fAxial = pow(rStar, expM) * pow(1.0 - rStar, expN);
220  // we use Hoekstra's method to calculate the fCirc based on fAxial
221  fCirc = fAxial * POD / pi / rPrime;
222  }
223  else
224  {
225  scalar dR2 = (rStar - rStarMax) * (rStar - rStarMax);
226  fAxial = fRMax * exp(-dR2 / epsRStar / epsRStar);
227  fCirc = fAxial * POD / pi / rPrimeMax;
228  }
229 
230  vector sourceVec = (fAxial * direction + fCirc * cellC2AVecCNorm);
231 
232  fvSource[cellI] += scale * etaAxial * etaTheta * sourceVec;
233 
234  thrustTotal += scale * etaAxial * etaTheta * fAxial * mesh_.V()[cellI];
235  torqueTotal += scale * etaAxial * etaTheta * fCirc * mesh_.V()[cellI];
236  }
237  reduce(thrustTotal, sumOp<scalar>());
238  reduce(torqueTotal, sumOp<scalar>());
239 
240  if (daOption_.getOption<word>("runStatus") == "solvePrimal")
241  {
242  if (mesh_.time().timeIndex() % printIntervalUnsteady_ == 0 || mesh_.time().timeIndex() == 1)
243  {
244  Info << "Actuator line source: " << lineName << endl;
245  Info << "Total thrust source: " << thrustTotal << endl;
246  Info << "Total torque source: " << torqueTotal << endl;
247  }
248  }
249  }
250 
251  fvSource.correctBoundaryConditions();
252 }
