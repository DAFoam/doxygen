60 void DAObjFuncStateErrorNorm::calcObjFunc(
61  const labelList& objFuncFaceSources,
62  const labelList& objFuncCellSources,
63  scalarList& objFuncFaceValues,
64  scalarList& objFuncCellValues,
65  scalar& objFuncValue)
66 {
67  /*
68  Description:
69  Calculate the stateErrorNorm
70  f = scale * L2Norm( state-stateRef )
71 
72  Input:
73  objFuncFaceSources: List of face source (index) for this objective
74 
75  objFuncCellSources: List of cell source (index) for this objective
76 
77  Output:
78  objFuncFaceValues: the discrete value of objective for each face source (index).
79  This will be used for computing df/dw in the adjoint.
80 
81  objFuncCellValues: the discrete value of objective on each cell source (index).
82  This will be used for computing df/dw in the adjoint.
83 
84  objFuncValue: the sum of objective, reduced across all processors and scaled by "scale"
85  */
86 
87  // initialize to zero
88  forAll(objFuncCellValues, idxI)
89  {
90  objFuncCellValues[idxI] = 0.0;
91  }
92  // initialize objFunValue
93  objFuncValue = 0.0;
94 
95  const objectRegistry& db = mesh_.thisDb();
96  if (stateType_ == "scalar")
97  {
98  const volScalarField& state = db.lookupObject<volScalarField>(stateName_);
99  const volScalarField& stateRef = db.lookupObject<volScalarField>(stateRefName_);
100  forAll(objFuncCellSources, idxI)
101  {
102  const label& cellI = objFuncCellSources[idxI];
103  objFuncCellValues[idxI] = scale_ * (sqr(state[cellI] - stateRef[cellI]));
104  objFuncValue += objFuncCellValues[idxI];
105  }
106  }
107  else if (stateType_ == "vector")
108  {
109  const volVectorField& state = db.lookupObject<volVectorField>(stateName_);
110  const volVectorField& stateRef = db.lookupObject<volVectorField>(stateRefName_);
111  forAll(objFuncCellSources, idxI)
112  {
113  const label& cellI = objFuncCellSources[idxI];
114  objFuncCellValues[idxI] = scale_ * (sqr(mag(state[cellI] - stateRef[cellI])));
115  objFuncValue += objFuncCellValues[idxI];
116  }
117  }
118 
119  // need to reduce the sum of force across all processors
120  reduce(objFuncValue, sumOp<scalar>());
121 
122  return;
123 }
