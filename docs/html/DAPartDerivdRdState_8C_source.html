73 void DAPartDerivdRdState::calcPartDerivMat(
74  const dictionary& options,
75  const Vec xvVec,
76  const Vec wVec,
77  Mat jacMat)
78 {
79  /*
80  Description:
81  Compute jacMat. We use analytical method
82 
83  Input:
84 
85  options.stateName: the name of the state in designVar
86 
87  xvVec: the volume mesh coordinate vector
88 
89  wVec: the state variable vector
90 
91  Output:
92  jacMat: the partial derivative matrix dRdState to compute
93  */
94 
95  // zero all the matrices
96  MatZeroEntries(jacMat);
97 
98  word stateNameInDesignVar = options.getWord("stateName");
99 
100  if (stateNameInDesignVar == "betaSA")
101  {
102 
103  scalarList prodTerm(mesh_.nCells());
104  daModel_.getTurbProdTerm(prodTerm);
105 
106  // dRdBetaSA only has diagonal component for the turbulence residual
107  forAll(mesh_.cells(), cellI)
108  {
109  label globalIndex = daIndex_.getGlobalAdjointStateIndex("nuTilda", cellI);
110  label globalCellIndex = daIndex_.getGlobalCellIndex(cellI);
111  // note the dR/dBetaSA is the negative of the prod term because
112  // the prod term shows up on the right hand side of the residual equation
113  scalar prodT = -prodTerm[cellI];
114  PetscScalar prodTermValue = 0.0;
115  assignValueCheckAD(prodTermValue, prodT);
116  MatSetValue(jacMat, globalIndex, globalCellIndex, prodTermValue, INSERT_VALUES);
117  }
118  }
119  else
120  {
121  FatalErrorIn("") << "stateName: " << stateNameInDesignVar << "not supported!"
122  << abort(FatalError);
123  }
124 
125  MatAssemblyBegin(jacMat, MAT_FINAL_ASSEMBLY);
126  MatAssemblyEnd(jacMat, MAT_FINAL_ASSEMBLY);
127 }
