50  const volScalarField chiByCv2((1 / Cv2_) * chi);

58 template<class BasicTurbulenceModel>
59 tmp<volScalarField> SpalartAllmarasFv3<BasicTurbulenceModel>::fw(
60  const volScalarField& Stilda) const
61 {
62  volScalarField r(
63  min(
64  nuTilda_
65  / (max(
66  Stilda,
67  dimensionedScalar("SMALL", Stilda.dimensions(), SMALL))
68  * sqr(kappa_ * y_)),
69  scalar(10.0)));
70  r.boundaryFieldRef() == 0.0;
71 
72  const volScalarField g(r + Cw2_ * (pow6(r) - r));
73 
74  return g * pow((1.0 + pow6(Cw3_)) / (pow6(g) + pow6(Cw3_)), 1.0 / 6.0);
75 }

253 void SpalartAllmarasFv3<BasicTurbulenceModel>::boundVar(volScalarField& var)
254 {
255  const scalar vGreat = 1e200;
256  label useUpperBound = 0, useLowerBound = 0;
257

259 
260  scalar varMin = simple.lookupOrDefault<scalar>(var.name() + "LowerBound", -vGreat);

262 
263  forAll(var, cellI)

268  useLowerBound = 1;
269  }
270  if (var[cellI] >= varMax)
271  {

273  useUpperBound = 1;

275  }

285 template<class BasicTurbulenceModel>
286 void SpalartAllmarasFv3<BasicTurbulenceModel>::correct()
287 {
288  if (!this->turbulence_)
289  {
290  return;
291  }
292 
293  // Local references
294  const alphaField& alpha = this->alpha_;
295  const rhoField& rho = this->rho_;
296  const surfaceScalarField& alphaRhoPhi = this->alphaRhoPhi_;
297  fv::options& fvOptions(fv::options::New(this->mesh_));
298 
299  eddyViscosity<RASModel<BasicTurbulenceModel>>::correct();
300 
301  const volScalarField chi(this->chi());
302  const volScalarField fv1(this->fv1(chi));
303 
304  const volScalarField Stilda(
305  fv3(chi, fv1) * ::sqrt(2.0) * mag(skew(fvc::grad(this->U_)))
306  + fv2(chi, fv1) * nuTilda_ / sqr(kappa_ * y_));
307 
308  tmp<fvScalarMatrix> nuTildaEqn(
309  fvm::ddt(alpha, rho, nuTilda_)
310  + fvm::div(alphaRhoPhi, nuTilda_)
311  - fvm::laplacian(alpha * rho * DnuTildaEff(), nuTilda_)
312  - Cb2_ / sigmaNut_ * alpha * rho * magSqr(fvc::grad(nuTilda_))
313  == Cb1_ * alpha * rho * Stilda * nuTilda_
314  - fvm::Sp(Cw1_ * alpha * rho * fw(Stilda) * nuTilda_ / sqr(y_), nuTilda_)
315  + fvOptions(alpha, rho, nuTilda_));
316 
317  nuTildaEqn.ref().relax();
318  fvOptions.constrain(nuTildaEqn.ref());
319  solve(nuTildaEqn);
320  fvOptions.correct(nuTilda_);
321  bound(nuTilda_, dimensionedScalar("0", nuTilda_.dimensions(), nuTildaMin_.value()));
322  boundVar(nuTilda_);
323  nuTilda_.correctBoundaryConditions();
324 
325  correctNut(fv1);
326 }
