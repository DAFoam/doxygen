56 void DAResidualSimpleFoam::calcResiduals(const dictionary& options)
57 {
58  /*
59  Description:
60  This is the function to compute residuals.
61 
62  Input:
63  options.isPC: 1 means computing residuals for preconditioner matrix.
64  This essentially use the first order scheme for div(phi,U)
65 
66  p_, U_, phi_, etc: State variables in OpenFOAM
67 
68  Output:
69  URes_, pRes_, phiRes_: residual field variables
70  */
71 
72  // We dont support MRF and fvOptions so all the related lines are commented
73  // out for now
74 
75  // ******** U Residuals **********
76  // copied and modified from UEqn.H
77 
78  word divUScheme = "div(phi,U)";
79 
80  label isPC = options.getLabel("isPC");
81 
82  if (isPC)
83  {
84  divUScheme = "div(pc)";
85  }
86 
87  tmp<fvVectorMatrix> tUEqn(
88  fvm::div(phi_, U_, divUScheme)
89  + daTurb_.divDevReff(U_)
90  - fvSource_);
91  fvVectorMatrix& UEqn = tUEqn.ref();
92 
93  UEqn.relax();
94 
95  URes_ = (UEqn & U_) + fvc::grad(p_);
96  normalizeResiduals(URes);
97 
98  // ******** p Residuals **********
99  // copied and modified from pEqn.H
100  // NOTE manually set pRefCell and pRefValue
101  label pRefCell = 0;
102  scalar pRefValue = 0.0;
103 
104  volScalarField rAU(1.0 / UEqn.A());
105  //volVectorField HbyA(constrainHbyA(rAU*UEqn.H(), U_, p_));
106  //***************** NOTE *******************
107  // constrainHbyA has been used since OpenFOAM-v1606; however, We do NOT use the constrainHbyA
108  // function in DAFoam because we found it significantly degrades the accuracy of shape derivatives.
109  // Basically, we should not constrain any variable because it will create discontinuity.
110  // Instead, we use the old implementation used in OpenFOAM-3.0+ and before
111  volVectorField HbyA("HbyA", U_);
112  HbyA = rAU * UEqn.H();
113 
114  surfaceScalarField phiHbyA("phiHbyA", fvc::flux(HbyA));
115 
116  adjustPhi(phiHbyA, U_, p_);
117 
118  tmp<volScalarField> rAtU(rAU);
119 
120  if (simple_.consistent())
121  {
122  rAtU = 1.0 / (1.0 / rAU - UEqn.H1());
123  phiHbyA += fvc::interpolate(rAtU() - rAU) * fvc::snGrad(p_) * mesh_.magSf();
124  HbyA -= (rAU - rAtU()) * fvc::grad(p_);
125  }
126 
127  tUEqn.clear();
128 
129  fvScalarMatrix pEqn(
130  fvm::laplacian(rAtU(), p_)
131  == fvc::div(phiHbyA));
132  pEqn.setReference(pRefCell, pRefValue);
133 
134  pRes_ = pEqn & p_;
135  normalizeResiduals(pRes);
136 
137  // ******** phi Residuals **********
138  // copied and modified from pEqn.H
139  phiRes_ = phiHbyA - pEqn.flux() - phi_;
140  // need to normalize phiRes
141  normalizePhiResiduals(phiRes);
142 }
