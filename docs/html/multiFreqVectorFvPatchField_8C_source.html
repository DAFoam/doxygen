94 void Foam::multiFreqVectorFvPatchField::updateCoeffs()
95 {
96  // calculate patch values
97  const scalar t = this->db().time().timeOutputValue();
98  scalar bcVal = 0.0;
99  const label& nFreqs = frequencies_.size();
100 
101  for (label i = 0; i < nFreqs; i++)
102  {
103  const scalar& a = amplitudes_[i];
104  const scalar& f = frequencies_[i];
105  const scalar& p = phases_[i];
106  bcVal += a * sin(constant::mathematical::twoPi * f * t + p);
107  }
108 
109  vectorField& thisPatchRef = *this;
110  vectorField thisPatch = thisPatchRef;
111  forAll(thisPatch, idxI) thisPatch[idxI] = refValue_;
112  if (t < endTime_) // if t > endTime, do not add oscillation.
113  {
114  forAll(thisPatch, idxI) thisPatch[idxI][component_] += bcVal;
115  }
116 
117  fvPatchVectorField::operator==(thisPatch);
118 
119  fixedValueFvPatchVectorField::updateCoeffs();
120 }
