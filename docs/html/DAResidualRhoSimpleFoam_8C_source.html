79 void DAResidualRhoSimpleFoam::calcResiduals(const dictionary& options)
80 {
81  /*
82  Description:
83  This is the function to compute residuals.
84 
85  Input:
86  options.isPC: 1 means computing residuals for preconditioner matrix.
87  This essentially use the first order scheme for div(phi,U), div(phi,e)
88 
89  p_, T_, U_, phi_, etc: State variables in OpenFOAM
90 
91  Output:
92  URes_, pRes_, TRes_, phiRes_: residual field variables
93  */
94 
95  // We dont support MRF and fvOptions so all the related lines are commented
96  // out for now
97 
98  label isPC = options.getLabel("isPC");
99 
100  word divUScheme = "div(phi,U)";
101  word divHEScheme = "div(phi,e)";
102 
103  if (he_.name() == "h")
104  {
105  divHEScheme = "div(phi,h)";
106  }
107 
108  if (isPC)
109  {
110  divUScheme = "div(pc)";
111  divHEScheme = "div(pc)";
112  }
113 
114  // ******** U Residuals **********
115  // copied and modified from UEqn.H
116 
117  tmp<fvVectorMatrix> tUEqn(
118  fvm::div(phi_, U_, divUScheme)
119  + daTurb_.divDevRhoReff(U_));
120  fvVectorMatrix& UEqn = tUEqn.ref();
121 
122  UEqn.relax();
123 
124  URes_ = (UEqn & U_) + fvc::grad(p_);
125  normalizeResiduals(URes);
126 
127  // ******** e Residuals **********
128  // copied and modified from EEqn.H
129  volScalarField alphaEff("alphaEff", thermo_.alphaEff(alphat_));
130 
131  fvScalarMatrix EEqn(
132  fvm::div(phi_, he_, divHEScheme)
133  + (he_.name() == "e"
134  ? fvc::div(phi_, volScalarField("Ekp", 0.5 * magSqr(U_) + p_ / rho_))
135  : fvc::div(phi_, volScalarField("K", 0.5 * magSqr(U_))))
136  - fvm::laplacian(alphaEff, he_));
137 
138  EEqn.relax();
139 
140  TRes_ = EEqn & he_;
141  normalizeResiduals(TRes);
142 
143  // ******** p and phi Residuals **********
144  // copied and modified from pEqn.H
145  volScalarField rAU(1.0 / UEqn.A());
146  surfaceScalarField rhorAUf("rhorAUf", fvc::interpolate(rho_ * rAU));
147  //volVectorField HbyA(constrainHbyA(rAU*UEqn.H(), U, p));
148  //***************** NOTE *******************
149  // constrainHbyA has been used since OpenFOAM-v1606; however, We do NOT use the constrainHbyA
150  // function in DAFoam because we found it significantly degrades the accuracy of shape derivatives.
151  // Basically, we should not constrain any variable because it will create discontinuity.
152  // Instead, we use the old implementation used in OpenFOAM-3.0+ and before
153  volVectorField HbyA("HbyA", U_);
154  HbyA = rAU * UEqn.H();
155  tUEqn.clear();
156 
157  surfaceScalarField phiHbyA("phiHbyA", fvc::interpolate(rho_) * fvc::flux(HbyA));
158 
159  // NOTE: we don't support transonic = true
160 
161  adjustPhi(phiHbyA, U_, p_);
162 
163  fvScalarMatrix pEqn(
164  fvc::div(phiHbyA)
165  - fvm::laplacian(rhorAUf, p_));
166 
167  pEqn.setReference(pressureControl_.refCell(), pressureControl_.refValue());
168 
169  pRes_ = pEqn & p_;
170  normalizeResiduals(pRes);
171 
172  // ******** phi Residuals **********
173  // copied and modified from pEqn.H
174  phiRes_ = phiHbyA + pEqn.flux() - phi_;
175  normalizePhiResiduals(phiRes);
176 }

178 void DAResidualRhoSimpleFoam::updateIntermediateVariables()
179 {
180  /*
181  Description:
182  Update the intermediate variables that depend on the state variables
183 
184  ********************** NOTE *****************
185  we assume hePsiThermo, pureMixture, perfectGas, hConst, and const transport
186  TODO: need to do this using built-in openfoam functions.
187 
188  we need to:
189  1, update psi based on T, psi=1/(R*T)
190  2, update rho based on p and psi, rho=psi*p
191  3, update E based on T, p and rho, E=Cp*T-p/rho
192  4, update velocity boundary based on MRF (not yet implemented)
193  */
194 
195  // 8314.4700665 gas constant in OpenFOAM
196  // src/OpenFOAM/global/constants/thermodynamic/thermodynamicConstants.H
197  scalar RR = Foam::constant::thermodynamic::RR;
198 
199  // R = RR/molWeight
200  // Foam::specie::R() function in src/thermophysicalModels/specie/specie/specieI.H
201  dimensionedScalar R(
202  "R1",
203  dimensionSet(0, 2, -2, -1, 0, 0, 0),
204  RR / molWeight_);
205 
206  // psi = 1/T/R
207  // see src/thermophysicalModels/specie/equationOfState/perfectGas/perfectGasI.H
208  psi_ = 1.0 / T_ / R;
209 
210  // rho = psi*p
211  // see src/thermophysicalModels/basic/psiThermo/psiThermo.C
212  rho_ = psi_ * p_;
213 
214  // **************** NOTE ****************
215  // need to relax rho to be consistent with the primal solver
216  // However, the rho.relax() will mess up perturbation
217  // That being said, we comment out the rho.relax() call to
218  // get the correct perturbed rho; however, the E residual will
219  // be a bit off compared with the ERes at the converged state
220  // from the primal solver. TODO. Need to figure out how to improve this
221  // **************** NOTE ****************
222  // rho_.relax();
223 
224  dimensionedScalar Cp(
225  "Cp1",
226  dimensionSet(0, 2, -2, -1, 0, 0, 0),
227  Cp_);
228 
229  // Hs = Cp*T
230  // see Hs() in src/thermophysicalModels/specie/thermo/hConst/hConstThermoI.H
231  // here the H departure EquationOfState::H(p, T) will be zero for perfectGas
232  // Es = Hs - p/rho = Hs - T * R;
233  // see Es() in src/thermophysicalModels/specie/thermo/thermo/thermoI.H
234  // **************** NOTE ****************
235  // See the comment from the rho.relax() call, if we write he_=Cp*T-p/rho, the
236  // accuracy of he_ may be impact by the inaccurate rho. So here we want to
237  // rewrite he_ as he_ = Cp * T_ - T_ * R instead, such that we dont include rho
238  // **************** NOTE ****************
239  if (he_.name() == "e")
240  {
241  he_ = Cp * T_ - T_ * R;
242  }
243  else
244  {
245  he_ = Cp * T_;
246  }
247  he_.correctBoundaryConditions();
248 
249  // NOTE: alphat is updated in the correctNut function in DATurbulenceModel child classes
250 }
