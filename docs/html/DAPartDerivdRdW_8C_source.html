73 void DAPartDerivdRdW::calcPartDerivMat(
74  const dictionary& options,
75  const Vec xvVec,
76  const Vec wVec,
77  Mat jacMat)
78 {
79  /*
80  Description:
81  Compute jacMat. We use coloring accelerated finite-difference
82 
83  Input:
84 
85  options.transposed. Whether to compute the transposed of dRdW
86 
87  options.isPC: whether to compute the jacMat for preconditioner
88 
89  options.lowerBound: any |value| that is smaller than lowerBound will be set to zero in dRdW
90 
91  xvVec: the volume mesh coordinate vector
92 
93  wVec: the state variable vector
94 
95  Output:
96  jacMat: the partial derivative matrix dRdW to compute
97  */
98 
99  label transposed = options.getLabel("transposed");
100 
101  // initialize coloredColumn vector
102  Vec coloredColumn;
103  VecDuplicate(wVec, &coloredColumn);
104  VecZeroEntries(coloredColumn);
105 
106  DAResidual& daResidual = const_cast<DAResidual&>(daResidual_);
107 
108  // zero all the matrices
109  MatZeroEntries(jacMat);
110 
111  Vec wVecNew;
112  VecDuplicate(wVec, &wVecNew);
113  VecCopy(wVec, wVecNew);
114 
115  // initialize residual vectors
116  Vec resVecRef, resVec;
117  VecDuplicate(wVec, &resVec);
118  VecDuplicate(wVec, &resVecRef);
119  VecZeroEntries(resVec);
120  VecZeroEntries(resVecRef);
121 
122  // set up state normalization vector
123  Vec normStatePerturbVec;
124  this->setNormStatePerturbVec(&normStatePerturbVec);
125 
126  dictionary mOptions;
127  mOptions.set("updateState", 1);
128  mOptions.set("updateMesh", 0);
129  mOptions.set("setResVec", 1);
130  mOptions.set("isPC", options.getLabel("isPC"));
131  daResidual.masterFunction(mOptions, xvVec, wVec, resVecRef);
132 
133  scalar jacLowerBound = options.getScalar("lowerBound");
134 
135  scalar delta = daOption_.getSubDictOption<scalar>("adjPartDerivFDStep", "State");
136  scalar rDelta = 1.0 / delta;
137 
138  label nColors = daJacCon_.getNJacConColors();
139 
140  word partDerivName = modelType_;
141  if (transposed)
142  {
143  partDerivName += "T";
144  }
145  if (options.getLabel("isPC"))
146  {
147  partDerivName += "PC";
148  }
149 
150  label printInterval = daOption_.getOption<label>("printInterval");
151  for (label color = 0; color < nColors; color++)
152  {
153  label eTime = mesh_.time().elapsedClockTime();
154  // print progress
155  if (color % printInterval == 0 or color == nColors - 1)
156  {
157  Info << partDerivName << ": " << color << " of " << nColors
158  << ", ExecutionTime: " << eTime << " s" << endl;
159  }
160 
161  // perturb states
162  this->perturbStates(
163  daJacCon_.getJacConColor(),
164  normStatePerturbVec,
165  color,
166  delta,
167  wVecNew);
168 
169  // compute residual
170  daResidual.masterFunction(mOptions, xvVec, wVecNew, resVec);
171 
172  // reset state perburbation
173  VecCopy(wVec, wVecNew);
174 
175  // compute residual partial using finite-difference
176  VecAXPY(resVec, -1.0, resVecRef);
177  VecScale(resVec, rDelta);
178 
179  // compute the colored coloumn and assign resVec to jacMat
180  daJacCon_.calcColoredColumns(color, coloredColumn);
181  this->setPartDerivMat(resVec, coloredColumn, transposed, jacMat, jacLowerBound);
182  }
183 
184  // call masterFunction again to reset the wVec to OpenFOAM field
185  daResidual.masterFunction(mOptions, xvVec, wVec, resVecRef);
186 
187  MatAssemblyBegin(jacMat, MAT_FINAL_ASSEMBLY);
188  MatAssemblyEnd(jacMat, MAT_FINAL_ASSEMBLY);
189 
190  if (daOption_.getOption<label>("debug"))
191  {
192  daIndex_.printMatChars(jacMat);
193  }
194 }
