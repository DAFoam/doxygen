509 tmp<volScalarField::Internal> DAkOmegaSSTLM::ReThetat0(
510  const volScalarField::Internal& Us,
511  const volScalarField::Internal& dUsds,
512  const volScalarField::Internal& nu) const
513 {
514 
515  tmp<volScalarField::Internal> tReThetat0(
516  new volScalarField::Internal(
517  IOobject(
518  IOobject::groupName("ReThetat0", U_.group()),
519  mesh_.time().timeName(),
520  mesh_),
521  mesh_,
522  dimless));
523  volScalarField::Internal& ReThetat0 = tReThetat0.ref();
524 
525  const volScalarField& k = k_;
526 
527  label maxIter = 0;
528 
529  forAll(ReThetat0, celli)
530  {
531  const scalar Tu(
532  max(100 * sqrt((2.0 / 3.0) * k[celli]) / Us[celli], scalar(0.027)));
533 
534  // Initialize lambda to zero.
535  // If lambda were cached between time-steps convergence would be faster
536  // starting from the previous time-step value.
537  scalar lambda = 0;
538 
539  scalar lambdaErr;
540  scalar thetat;
541  label iter = 0;
542 
543  do
544  {
545  // Previous iteration lambda for convergence test
546  const scalar lambda0 = lambda;
547 
548  if (Tu <= 1.3)
549  {
550  const scalar Flambda =
551  dUsds[celli] <= 0
552  ? scalar(1
553  - (-12.986 * lambda
554  - 123.66 * sqr(lambda)
555  - 405.689 * pow3(lambda))
556  * exp(-pow(Tu / 1.5, 1.5)))
557  : scalar(1
558  + 0.275 * (1 - exp(-35 * lambda))
559  * exp(-Tu / 0.5));
560 
561  thetat =
562  (1173.51 - 589.428 * Tu + 0.2196 / sqr(Tu))
563  * Flambda * nu[celli]
564  / Us[celli];
565  }
566  else
567  {
568  const scalar Flambda =
569  dUsds[celli] <= 0
570  ? scalar(1
571  - (-12.986 * lambda
572  - 123.66 * sqr(lambda)
573  - 405.689 * pow3(lambda))
574  * exp(-pow(Tu / 1.5, 1.5)))
575  : scalar(1
576  + 0.275 * (1 - exp(-35 * lambda))
577  * exp(-2 * Tu));
578 
579  thetat =
580  331.50 * pow((Tu - 0.5658), -0.671)
581  * Flambda * nu[celli] / Us[celli];
582  }
583 
584  lambda = sqr(thetat) / nu[celli] * dUsds[celli];
585  lambda = max(min(lambda, 0.1), -0.1);
586 
587  lambdaErr = mag(lambda - lambda0);
588 
589  maxIter = max(maxIter, ++iter);
590 
591  } while (lambdaErr > lambdaErr_);
592 
593  ReThetat0[celli] = max(thetat * Us[celli] / nu[celli], scalar(20));
594  }
595 
596  if (maxIter > maxLambdaIter_)
597  {
598  WarningInFunction
599  << "Number of lambda iterations exceeds maxLambdaIter("
600  << maxLambdaIter_ << ')' << endl;
601  }
602 
603  return tReThetat0;
604 }

612

646 {

682 void DAkOmegaSSTLM::correctOmegaBoundaryConditions()
683 {
684  /*
685  Description:
686  this is a special treatment for omega BC because we cant directly call omega.
687  correctBoundaryConditions() because it will modify the internal omega and G that
688  are right next to walls. This will mess up adjoint Jacobians
689  To solve this issue,
690  1. we store the near wall omega before calling omega.correctBoundaryConditions()
691  2. call omega.correctBoundaryConditions()
692  3. Assign the stored near wall omega back
693  4. Apply a zeroGradient BC for omega at the wall patches
694  *********** NOTE *************
695  this treatment will obviously downgrade the accuracy of adjoint derivative since it is
696  not 100% consistent with what is used for the flow solver; however, our observation
697  shows that the impact is not very large.
698  */
699 
700  // save the perturbed omega at the wall
701  this->saveOmegaNearWall();
702  // correct omega boundary conditions, this includes updating wall face and near wall omega values,
703  // updating the inter-proc BCs
704  omega_.correctBoundaryConditions();
705  // reset the corrected omega near wall cell to its perturbed value
706  this->setOmegaNearWall();
707 }

773 void DAkOmegaSSTLM::correctStateResidualModelCon(List<List<word>>& stateCon) const
774 {
775  /*
776  Description:
777  Update the original variable connectivity for the adjoint state
778  residuals in stateCon. Basically, we modify/add state variables based on the
779  original model variables defined in stateCon.
780 
781  Input:
782 
783  stateResCon: the connectivity levels for a state residual, defined in Foam::DAJacCon
784 
785  Example:
786  If stateCon reads:
787  stateCon=
788  {
789  {"U", "p", "nut"},
790  {"p"}
791  }
792 
793  For the SA turbulence model, calling this function for will get a new stateCon
794  stateCon=
795  {
796  {"U", "p", "nuTilda"},
797  {"p"}
798  }
799 
800  For the SST turbulence model, calling this function will give
801  stateCon=
802  {
803  {"U", "p", "k", "omega"},
804  {"p", "U"}
805  }
806  ***NOTE***: we add a extra level of U connectivity because nut is
807  related to grad(U), k, and omega in SST!
808  */
809 
810  label stateConSize = stateCon.size();
811  forAll(stateCon, idxI)
812  {
813  label addUCon = 0;
814  forAll(stateCon[idxI], idxJ)
815  {
816  word conStateName = stateCon[idxI][idxJ];
817  if (conStateName == "nut")
818  {
819  stateCon[idxI][idxJ] = "omega"; // replace nut with omega
820  stateCon[idxI].append("k"); // also add k for that level
821  stateCon[idxI].append("ReThetat");
822  stateCon[idxI].append("gammaInt");
823  addUCon = 1;
824  }
825  }
826  // add U for the current level and level+1 if it is not there yet
827  label isU;
828  if (addUCon == 1)
829  {
830  // first add U for the current level
831  isU = 0;
832  forAll(stateCon[idxI], idxJ)
833  {
834  word conStateName = stateCon[idxI][idxJ];
835  if (conStateName == "U")
836  {
837  isU = 1;
838  }
839  }
840  if (!isU)
841  {
842  stateCon[idxI].append("U");
843  }
844 
845  // now add U for level+1 if idxI is not the largest level
846  // if idxI is already the largest level, we have a problem
847  if (idxI != stateConSize - 1)
848  {
849  isU = 0;
850  forAll(stateCon[idxI + 1], idxJ)
851  {
852  word conStateName = stateCon[idxI + 1][idxJ];
853  if (conStateName == "U")
854  {
855  isU = 1;
856  }
857  }
858  if (!isU)
859  {
860  stateCon[idxI + 1].append("U");
861  }
862  }
863  else
864  {
865  FatalErrorIn(
866  "In DAStateInfo, nut shows in the largest connectivity level! "
867  "This is not supported!")
868  << exit(FatalError);
869  }
870  }
871  }

998 void DAkOmegaSSTLM::correct()
999 {
1000  /*
1001  Descroption:
1002  Solve the residual equations and update the state. This function will be called
1003  by the DASolver. It is needed because we want to control the output frequency
1004  of the residual convergence every 100 steps. If using the correct from turbulence
1005  it will output residual every step which will be too much of information.
1006  */
1007 
1008  // We set the flag solveTurbState_ to 1 such that in the calcResiduals function
1009  // we will solve and update nuTilda
1010  solveTurbState_ = 1;
1011  dictionary dummyOptions;
1012  this->calcResiduals(dummyOptions);
1013  // after it, we reset solveTurbState_ = 0 such that calcResiduals will not
1014  // update nuTilda when calling from the adjoint class, i.e., solveAdjoint from DASolver.
1015  solveTurbState_ = 0;
1016 }

1018 void DAkOmegaSSTLM::calcResiduals(const dictionary& options)
1019 {
1020  /*
1021  Descroption:
1022  If solveTurbState_ == 1, this function solve and update k and omega, and
1023  is the same as calling turbulence.correct(). If solveTurbState_ == 0,
1024  this function compute residuals for turbulence variables, e.g., nuTildaRes_
1025 
1026  Input:
1027  options.isPC: 1 means computing residuals for preconditioner matrix.
1028  This essentially use the first order scheme for div(phi,nuTilda)
1029 
1030  p_, U_, phi_, etc: State variables in OpenFOAM
1031 
1032  Output:
1033  kRes_/omegaRes_: If solveTurbState_ == 0, update the residual field variable
1034 
1035  k_/omega_: If solveTurbState_ == 1, update them
1036  */
1037 
1038  // Copy and modify based on the "correct" function
1039 
1040  label printToScreen = this->isPrintTime(mesh_.time(), printInterval_);
1041 
1042  word divKScheme = "div(phi,k)";
1043  word divOmegaScheme = "div(phi,omega)";
1044  word divReThetatScheme = "div(phi,ReThetat)";
1045  word divGammaIntScheme = "div(phi,gammaInt)";
1046 
1047  label isPC = 0;
1048 
1049  if (!solveTurbState_)
1050  {
1051  isPC = options.getLabel("isPC");
1052 
1053  if (isPC)
1054  {
1055  divKScheme = "div(pc)";
1056  divOmegaScheme = "div(pc)";
1057  divReThetatScheme = "div(pc)";
1058  divGammaIntScheme = "div(pc)";
1059  }
1060  }
1061  // ************ SST part ********
1062  {
1063  // Note: for compressible flow, the "this->phi()" function divides phi by fvc:interpolate(rho),
1064  // while for the incompresssible "this->phi()" returns phi only
1065  // see src/TurbulenceModels/compressible/compressibleTurbulenceModel.C line 62 to 73
1066  volScalarField::Internal divU(fvc::div(fvc::absolute(phi_ / fvc::interpolate(rho_), U_)));
1067 
1068  tmp<volTensorField> tgradU = fvc::grad(U_);
1069  volScalarField S2(2 * magSqr(symm(tgradU())));
1070  volScalarField::Internal GbyNu0((tgradU() && dev(twoSymm(tgradU()))));
1071  volScalarField::Internal G("kOmegaSSTLM:G", nut_ * GbyNu0);
1072 
1073  if (solveTurbState_)
1074  {
1075  // Update omega and G at the wall
1076  omega_.boundaryFieldRef().updateCoeffs();
1077  }
1078  else
1079  {
1080  // NOTE instead of calling omega_.boundaryFieldRef().updateCoeffs();
1081  // here we call our self-defined boundary conditions
1082  this->correctOmegaBoundaryConditions();
1083  }
1084 
1085  volScalarField CDkOmega(
1086  (scalar(2) * alphaOmega2_) * (fvc::grad(k_) & fvc::grad(omega_)) / omega_);
1087 
1088  volScalarField F1(this->F1(CDkOmega));
1089  volScalarField F23(this->F23());
1090 
1091  {
1092 
1093  volScalarField::Internal gamma(this->gamma(F1));
1094  volScalarField::Internal beta(this->beta(F1));
1095 
1096  // Turbulent frequency equation
1097  tmp<fvScalarMatrix> omegaEqn(
1098  fvm::ddt(phase_, rho_, omega_)
1099  + fvm::div(phaseRhoPhi_, omega_, divOmegaScheme)
1100  - fvm::laplacian(phase_ * rho_ * DomegaEff(F1), omega_)
1101  == phase_() * rho_() * gamma * GbyNu(GbyNu0, F23(), S2())
1102  - fvm::SuSp((2.0 / 3.0) * phase_() * rho_() * gamma * divU, omega_)
1103  - fvm::Sp(phase_() * rho_() * beta * omega_(), omega_)
1104  - fvm::SuSp(
1105  phase_() * rho_() * (F1() - scalar(1)) * CDkOmega() / omega_(),
1106  omega_)
1107  + Qsas(S2(), gamma, beta)
1108  + omegaSource()
1109 
1110  );
1111 
1112  omegaEqn.ref().relax();
1113  omegaEqn.ref().boundaryManipulate(omega_.boundaryFieldRef());
1114 
1115  if (solveTurbState_)
1116  {
1117 
1118  // get the solver performance info such as initial
1119  // and final residuals
1120  SolverPerformance<scalar> solverOmega = solve(omegaEqn);
1121  if (printToScreen)
1122  {
1123  Info << "omega Initial residual: " << solverOmega.initialResidual() << endl
1124  << " Final residual: " << solverOmega.finalResidual() << endl;
1125  }
1126 
1127  DAUtility::boundVar(allOptions_, omega_, printToScreen);
1128  }
1129  else
1130  {
1131  // reset the corrected omega near wall cell to its perturbed value
1132  this->setOmegaNearWall();
1133 
1134  // calculate residuals
1135  omegaRes_ = omegaEqn() & omega_;
1136  // need to normalize residuals
1137  normalizeResiduals(omegaRes);
1138  }
1139  }
1140 
1141  // Turbulent kinetic energy equation
1142  tmp<fvScalarMatrix> kEqn(
1143  fvm::ddt(phase_, rho_, k_)
1144  + fvm::div(phaseRhoPhi_, k_, divKScheme)
1145  - fvm::laplacian(phase_ * rho_ * DkEff(F1), k_)
1146  == phase_() * rho_() * Pk(G)
1147  - fvm::SuSp((2.0 / 3.0) * phase_() * rho_() * divU, k_)
1148  - fvm::Sp(phase_() * rho_() * epsilonByk(F1, tgradU()), k_)
1149  + kSource());
1150 
1151  tgradU.clear();
1152 
1153  kEqn.ref().relax();
1154 
1155  if (solveTurbState_)
1156  {
1157 
1158  // get the solver performance info such as initial
1159  // and final residuals
1160  SolverPerformance<scalar> solverK = solve(kEqn);
1161  if (printToScreen)
1162  {
1163  Info << "k Initial residual: " << solverK.initialResidual() << endl
1164  << " Final residual: " << solverK.finalResidual() << endl;
1165  }
1166 
1167  DAUtility::boundVar(allOptions_, k_, printToScreen);
1168 
1169  this->correctNut();
1170  }
1171  else
1172  {
1173  // calculate residuals
1174  kRes_ = kEqn() & k_;
1175  // need to normalize residuals
1176  normalizeResiduals(kRes);
1177  }
1178  }
1179 
1180  // ************ LM part ********
1181  {
1182  // we need to bound before computing residuals
1183  // this will avoid having NaN residuals
1184  DAUtility::boundVar(allOptions_, ReThetat_, printToScreen);
1185 
1186  // Local references
1187  const tmp<volScalarField> tnu = this->nu();
1188  const volScalarField::Internal& nu = tnu()();
1189  const volScalarField::Internal& y = y_();
1190 
1191  // Fields derived from the velocity gradient
1192  tmp<volTensorField> tgradULM = fvc::grad(U_);
1193  const volScalarField::Internal Omega(sqrt(2 * magSqr(skew(tgradULM()()))));
1194  const volScalarField::Internal S(sqrt(2 * magSqr(symm(tgradULM()()))));
1195  const volScalarField::Internal Us(max(mag(U_()), deltaU_));
1196  const volScalarField::Internal dUsds((U_() & (U_() & tgradULM()())) / sqr(Us));
1197  tgradULM.clear();
1198 
1199  const volScalarField::Internal Fthetat(this->Fthetat(Us, Omega, nu));
1200 
1201  {
1202  const volScalarField::Internal t(500 * nu / sqr(Us));
1203  const volScalarField::Internal Pthetat(
1204  phase_() * rho_() * (cThetat_ / t) * (1 - Fthetat));
1205 
1206  // Transition onset momentum-thickness Reynolds number equation
1207  tmp<fvScalarMatrix> ReThetatEqn(
1208  fvm::ddt(phase_, rho_, ReThetat_)
1209  + fvm::div(phaseRhoPhi_, ReThetat_, divReThetatScheme)
1210  - fvm::laplacian(phase_ * rho_ * DReThetatEff(), ReThetat_)
1211  == Pthetat * ReThetat0(Us, dUsds, nu) - fvm::Sp(Pthetat, ReThetat_));
1212 
1213  ReThetatEqn.ref().relax();
1214 
1215  if (solveTurbState_)
1216  {
1217 
1218  // get the solver performance info such as initial
1219  // and final residuals
1220  SolverPerformance<scalar> solverReThetat = solve(ReThetatEqn);
1221  if (printToScreen)
1222  {
1223  Info << "ReThetat Initial residual: " << solverReThetat.initialResidual() << endl
1224  << " Final residual: " << solverReThetat.finalResidual() << endl;
1225  }
1226 
1227  DAUtility::boundVar(allOptions_, ReThetat_, printToScreen);
1228  }
1229  else
1230  {
1231  // calculate residuals
1232  ReThetatRes_ = ReThetatEqn() & ReThetat_;
1233  // need to normalize residuals
1234  normalizeResiduals(ReThetatRes);
1235  }
1236  }
1237 
1238  // we need to bound before computing residuals
1239  // this will avoid having NaN residuals
1240  DAUtility::boundVar(allOptions_, gammaInt_, printToScreen);
1241 
1242  const volScalarField::Internal ReThetac(this->ReThetac());
1243  const volScalarField::Internal Rev(sqr(y) * S / nu);
1244  const volScalarField::Internal RT(k_() / (nu * omega_()));
1245 
1246  {
1247  const volScalarField::Internal Pgamma(
1248  phase_() * rho_()
1249  * ca1_ * Flength(nu) * S * sqrt(gammaInt_() * Fonset(Rev, ReThetac, RT)));
1250 
1251  const volScalarField::Internal Fturb(exp(-pow4(0.25 * RT)));
1252 
1253  const volScalarField::Internal Egamma(
1254  phase_() * rho_() * ca2_ * Omega * Fturb * gammaInt_());
1255 
1256  // Intermittency equation
1257  tmp<fvScalarMatrix> gammaIntEqn(
1258  fvm::ddt(phase_, rho_, gammaInt_)
1259  + fvm::div(phaseRhoPhi_, gammaInt_, divGammaIntScheme)
1260  - fvm::laplacian(phase_ * rho_ * DgammaIntEff(), gammaInt_)
1261  == Pgamma - fvm::Sp(ce1_ * Pgamma, gammaInt_)
1262  + Egamma - fvm::Sp(ce2_ * Egamma, gammaInt_));
1263 
1264  gammaIntEqn.ref().relax();
1265 
1266  if (solveTurbState_)
1267  {
1268 
1269  // get the solver performance info such as initial
1270  // and final residuals
1271  SolverPerformance<scalar> solverGammaInt = solve(gammaIntEqn);
1272  if (printToScreen)
1273  {
1274  Info << "gammaInt Initial residual: " << solverGammaInt.initialResidual() << endl
1275  << " Final residual: " << solverGammaInt.finalResidual() << endl;
1276  }
1277 
1278  DAUtility::boundVar(allOptions_, gammaInt_, printToScreen);
1279 
1280  const volScalarField::Internal Freattach(exp(-pow4(RT / 20.0)));
1281  const volScalarField::Internal gammaSep(
1282  min(2 * max(Rev / (3.235 * ReThetac) - 1, scalar(0)) * Freattach, scalar(2))
1283  * Fthetat);
1284 
1285  gammaIntEff_ = max(gammaInt_(), gammaSep);
1286  }
1287  else
1288  {
1289  // calculate residuals
1290  gammaIntRes_ = gammaIntEqn() & gammaInt_;
1291  // need to normalize residuals
1292  normalizeResiduals(gammaIntRes);
1293  }
1294  }
1295  }
1296 
1297  return;
1298 }
