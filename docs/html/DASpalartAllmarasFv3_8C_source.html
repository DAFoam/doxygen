133 tmp<volScalarField> DASpalartAllmarasFv3::fv3(
134  const volScalarField& chi,
135  const volScalarField& fv1) const
136 {
137 
138  const volScalarField chiByCv2((1 / Cv2_) * chi);
139 
140  return (scalar(1) + chi * fv1)
141  * (1 / Cv2_)
142  * (3 * (scalar(1) + chiByCv2) + sqr(chiByCv2))
143  / pow3(scalar(1) + chiByCv2);

146 tmp<volScalarField> DASpalartAllmarasFv3::fw(
147  const volScalarField& Stilda) const
148 {
149  volScalarField r(
150  min(
151  nuTilda_
152  / (max(
153  Stilda,
154  dimensionedScalar("SMALL", Stilda.dimensions(), SMALL))
155  * sqr(kappa_ * y_)),
156  scalar(10.0)));
157  r.boundaryFieldRef() == 0.0;
158 
159  const volScalarField g(r + Cw2_ * (pow6(r) - r));
160 
161  return g * pow((1.0 + pow6(Cw3_)) / (pow6(g) + pow6(Cw3_)), 1.0 / 6.0);
162 }

375 void DASpalartAllmarasFv3::correct()
376 {
377  /*
378  Descroption:
379  Solve the residual equations and update the state. This function will be called
380  by the DASolver. It is needed because we want to control the output frequency
381  of the residual convergence every 100 steps. If using the correct from turbulence
382  it will output residual every step which will be too much of information.
383  */
384 
385  // We set the flag solveTurbState_ to 1 such that in the calcResiduals function
386  // we will solve and update nuTilda
387  solveTurbState_ = 1;
388  dictionary dummyOptions;
389  this->calcResiduals(dummyOptions);
390  // after it, we reset solveTurbState_ = 0 such that calcResiduals will not
391  // update nuTilda when calling from the adjoint class, i.e., solveAdjoint from DASolver.
392  solveTurbState_ = 0;
393 }

395 void DASpalartAllmarasFv3::calcResiduals(const dictionary& options)
396 {
397  /*
398  Descroption:
399  If solveTurbState_ == 1, this function solve and update nuTilda, and
400  is the same as calling turbulence.correct(). If solveTurbState_ == 0,
401  this function compute residuals for turbulence variables, e.g., nuTildaRes_
402 
403  Input:
404  options.isPC: 1 means computing residuals for preconditioner matrix.
405  This essentially use the first order scheme for div(phi,nuTilda)
406 
407  p_, U_, phi_, etc: State variables in OpenFOAM
408 
409  Output:
410  nuTildaRes_: If solveTurbState_ == 0, update the residual field variable
411 
412  nuTilda_: If solveTurbState_ == 1, update nuTilda
413  */
414 
415  // Copy and modify based on the "correct" function
416 
417  label printToScreen = this->isPrintTime(mesh_.time(), printInterval_);
418 
419  word divNuTildaScheme = "div(phi,nuTilda)";
420 
421  label isPC = 0;
422 
423  if (!solveTurbState_)
424  {
425  isPC = options.getLabel("isPC");
426 
427  if (isPC)
428  {
429  divNuTildaScheme = "div(pc)";
430  }
431  }
432 
433  const volScalarField chi(this->chi());
434  const volScalarField fv1(this->fv1(chi));
435 
436  const volScalarField Stilda(
437  this->fv3(chi, fv1) * ::sqrt(2.0) * mag(skew(fvc::grad(U_)))
438  + this->fv2(chi, fv1) * nuTilda_ / sqr(kappa_ * y_));
439 
440  tmp<fvScalarMatrix> nuTildaEqn(
441  fvm::ddt(phase_, rho_, nuTilda_)
442  + fvm::div(phaseRhoPhi_, nuTilda_, divNuTildaScheme)
443  - fvm::laplacian(phase_ * rho_ * DnuTildaEff(), nuTilda_)
444  - Cb2_ / sigmaNut_ * phase_ * rho_ * magSqr(fvc::grad(nuTilda_))
445  == Cb1_ * phase_ * rho_ * Stilda * nuTilda_
446  - fvm::Sp(Cw1_ * phase_ * rho_ * fw(Stilda) * nuTilda_ / sqr(y_), nuTilda_));
447 
448  nuTildaEqn.ref().relax();
449 
450  if (solveTurbState_)
451  {
452 
453  // get the solver performance info such as initial
454  // and final residuals
455  SolverPerformance<scalar> solverNuTilda = solve(nuTildaEqn);
456  if (printToScreen)
457  {
458  Info << "nuTilda Initial residual: " << solverNuTilda.initialResidual() << endl
459  << " Final residual: " << solverNuTilda.finalResidual() << endl;
460  }
461 
462  DAUtility::boundVar(allOptions_, nuTilda_, printToScreen);
463  nuTilda_.correctBoundaryConditions();
464 
465  // ***************** NOTE*****************
466  // In the original SA, it is correctNut(fv1) and fv1 is not
467  // updated based on the latest nuTilda. We use correctNut which
468  // recompute fv1 with the latest nuTilda
469  this->correctNut();
470  }
471  else
472  {
473  // calculate residuals
474  nuTildaRes_ = nuTildaEqn.ref() & nuTilda_;
475  // need to normalize residuals
476  normalizeResiduals(nuTildaRes);
477  }
478 
479  return;
480 }
