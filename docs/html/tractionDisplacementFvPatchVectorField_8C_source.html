98 void tractionDisplacementFvPatchVectorField::updateCoeffs()
99 {
100  if (updated())
101  {
102  return;
103  }
104 
105  const dictionary& mechanicalProperties =
106  db().lookupObject<IOdictionary>("mechanicalProperties");
107 
108  const fvPatchField<scalar>& rho =
109  patch().lookupPatchField<volScalarField, scalar>("solid:rho");
110 
111  const fvPatchField<scalar>& rhoE =
112  patch().lookupPatchField<volScalarField, scalar>("E");
113 
114  const fvPatchField<scalar>& nu =
115  patch().lookupPatchField<volScalarField, scalar>("solid:nu");
116 
117  scalarField E(rhoE / rho);
118  scalarField mu(E / (2.0 * (1.0 + nu)));
119  scalarField lambda(nu * E / ((1.0 + nu) * (1.0 - 2.0 * nu)));
120 
121  Switch planeStress(mechanicalProperties.lookup("planeStress"));
122 
123  if (planeStress)
124  {
125  lambda = nu * E / ((1.0 + nu) * (1.0 - nu));
126  }
127 
128  scalarField twoMuLambda(2 * mu + lambda);
129 
130  vectorField n(patch().nf());
131 
132  // NOTE: we can't use the built-in gradient() computation because
133  // it is designed for transient problem, i.e., snGrad() is actually
134  // related to gradient()
135  // Here we implement our owned BC for steady-state
136  const fvPatchField<tensor>& gradD =
137  patch().lookupPatchField<volTensorField, tensor>("gradD");
138  gradient() = ((traction_ - pressure_ * n) / rho
139  - (n & (mu * gradD.T() - (mu + lambda) * gradD))
140  - n * tr(gradD) * lambda)
141  / (2.0 * mu + lambda);
142 
143  fixedGradientFvPatchVectorField::updateCoeffs();
144 }
