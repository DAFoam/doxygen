156 label DAUtility::listReplaceVal(
157  List<classType>& listIn,
158  const classType valOrig,
159  const classType valNew)
160 {
161  /*
162  Description:
163  Replace a value in hte list, the list sequence is preserved
164 
165  Input:
166  listIn: the list to replace
167  valOrig: original value in the list
168  valNew: new value to replace
169 
170  Example:
171  List<word> listIn={"apple","orange"};
172  listReplaceVal<word>(listIn,"apple","banana");
173  Now listIn will be {"banana","orange"}
174  NOTE: if list has multiple valOrig, they will be all replaced
175  */
176 
177  label foundVal = 0;
178  forAll(listIn, idxI)
179  {
180  const classType& val = listIn[idxI];
181  if (val == valOrig)
182  {
183  listIn[idxI] = valNew;
184  foundVal = 1;
185  }
186  }
187 
188  return foundVal;
189 }

192 label DAUtility::listDeleteVal(
193  List<classType>& listIn,
194  const classType valDel)
195 {
196  /*
197  Description:
198  Delete a value in the list, the sequence of other
199  elements will be preserved
200 
201  Input:
202  listIn: the list to delete value
203  valDel: value to delete
204 
205  Output:
206  foundVal: 1 means the requested value is found and deleted
207 
208  Example:
209  List<word> listIn={"apple","orange"};
210  listDeleteVal<word>(listIn,"apple");
211  Now listIn will be {orange"}
212  NOTE: if list has multiple val, they will be all deleted
213  */
214 
215  // we will create a new list and delete the value
216  List<classType> listNew;
217 
218  label foundVal = 0;
219  forAll(listIn, idxI)
220  {
221  const classType& val = listIn[idxI];
222  if (val == valDel)
223  {
224  foundVal = 1;
225  // do not append
226  }
227  else
228  {
229  listNew.append(val);
230  }
231  }
232 
233  listIn.clear();
234  listIn = listNew;
235 
236  return foundVal;
237 }
