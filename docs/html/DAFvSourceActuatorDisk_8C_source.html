35 void DAFvSourceActuatorDisk::calcFvSource(volVectorField& fvSource)
36 {
37  /*
38  Description:
39  Compute the actuator disk source term.
40  We follow: Hoekstra, A RANS-based analysis tool for ducted propeller systems
41  in open water condition, International Shipbuilding Progress
42  source = rStar * Foam::sqrt(1.0 - rStar) * scale
43  where rStar is the normalized radial location and scale is used to
44  make sure the integral force equals the desired total thrust
45 
46  Example:
47  An example of the fvSource in pyOptions in pyDAFoam can be
48  defOptions =
49  {
50  "fvSource"
51  {
52  "disk1"
53  {
54  "type": "actuatorDisk",
55  "source": cylinderAnnulusToCell,
56  "p1": [0.5, 0.3, 0.1], # p1 and p2 define the axis and width
57  "p2": [0.5, 0.3, 0.5], # p2-p1 should be streamwise
58  "innerRadius": 0.1,
59  "outerRadius": 0.8,
60  "rotDir": "left",
61  "scale": 1.0,
62  "POD": 0.7
63  },
64  "disk2"
65  {
66  "type": "actuatorDisk",
67  "source": cylinderAnnulusToCell,
68  "p1": [0.0, 0.0, 0.1],
69  "p2": [0.0, 0.0, 0.5],
70  "innerRadius": 0.1,
71  "outerRadius": 0.8,
72  "rotDir": "right",
73  "scale": 0.1,
74  "POD": 1.0
75  }
76  }
77  }
78  */
79 
80  forAll(fvSource, idxI)
81  {
82  fvSource[idxI] = vector::zero;
83  }
84 
85  // loop over all the cell indices for all actuator disks
86  forAll(fvSourceCellIndices_.toc(), idxI)
87  {
88 
89  // name of this disk
90  word diskName = fvSourceCellIndices_.toc()[idxI];
91 
92  // sub dictionary with all parameters for this disk
93  dictionary diskSubDict = fvSourceSubDict_.subDict(diskName);
94 
95  // now read in all parameters for this actuator disk
96  scalarList point1;
97  scalarList point2;
98  diskSubDict.readEntry<scalarList>("p1", point1);
99  diskSubDict.readEntry<scalarList>("p2", point2);
100  vector p1 = {point1[0], point1[1], point1[2]};
101  vector p2 = {point2[0], point2[1], point2[2]};
102  vector diskCenter = (p1 + p2) / 2.0;
103  vector diskDir = p2 - p1; // NOTE: p2 - p1 should be streamwise
104  vector diskDirNorm = diskDir / mag(diskDir);
105  scalar outerRadius = diskSubDict.getScalar("outerRadius");
106  scalar innerRadius = diskSubDict.getScalar("innerRadius");
107  word rotDir = diskSubDict.getWord("rotDir");
108  scalar scale = diskSubDict.getScalar("scale");
109  scalar POD = diskSubDict.getScalar("POD");
110 
111  // loop over all cell indices for this disk and computer the source term
112  scalar thrustSourceSum = 0.0;
113  scalar torqueSourceSum = 0.0;
114  forAll(fvSourceCellIndices_[diskName], idxJ)
115  {
116  // cell index
117  label cellI = fvSourceCellIndices_[diskName][idxJ];
118 
119  // the cell center coordinates of this cellI
120  vector cellC = mesh_.C()[cellI];
121  // cell center to disk center vector
122  vector cellC2AVec = cellC - diskCenter;
123  // tmp tensor for calculating the axial/radial components of cellC2AVec
124  tensor cellC2AVecE(tensor::zero);
125  cellC2AVecE.xx() = cellC2AVec.x();
126  cellC2AVecE.yy() = cellC2AVec.y();
127  cellC2AVecE.zz() = cellC2AVec.z();
128 
129  // now we need to decompose cellC2AVec into axial and radial components
130  // the axial component of cellC2AVec vector
131  vector cellC2AVecA = cellC2AVecE & diskDirNorm;
132  // the radial component of cellC2AVec vector
133  vector cellC2AVecR = cellC2AVec - cellC2AVecA;
134 
135  // now we can use the cross product to compute the tangential
136  // (circ) direction of cellI
137  vector cellC2AVecC(vector::zero);
138  if (rotDir == "left")
139  {
140  // this assumes right hand rotation of propellers
141  cellC2AVecC = cellC2AVecR ^ cellC2AVecA; // circ
142  }
143  else if (rotDir == "right")
144  {
145  // this assumes left hand rotation of propellers
146  cellC2AVecC = cellC2AVecA ^ cellC2AVecR; // circ
147  }
148  else
149  {
150  FatalErrorIn(" ") << "rotDir not valid" << abort(FatalError);
151  }
152 
153  // the magnigude of radial compoent of cellC2AVecR
154  scalar cellC2AVecRLen = mag(cellC2AVecR);
155  // the magnigude of tangential compoent of cellC2AVecR
156  scalar cellC2AVecCLen = mag(cellC2AVecC);
157  // the normalized cellC2AVecC (tangential) vector
158  vector cellC2AVecCNorm = cellC2AVecC / cellC2AVecCLen;
159 
160  // now we can use Hoekstra's formulation to compute source
161  scalar rPrime = cellC2AVecRLen / outerRadius;
162  scalar rPrimeHub = innerRadius / outerRadius;
163  // rStar is normalized radial location
164  scalar rStar = (rPrime - rPrimeHub) / (1.0 - rPrimeHub);
165  // axial force, NOTE: user need to prescribe "scale" such that the integraged
166  // axial force matches the desired thrust
167  scalar fAxial = rStar * Foam::sqrt(1.0 - rStar) * scale;
168  // we use Hoekstra's method to calculate the fCirc based on fAxial
169  scalar fCirc = fAxial * POD / constant::mathematical::pi / rPrime;
170  vector sourceVec = (fAxial * diskDirNorm + fCirc * cellC2AVecCNorm);
171  // the source is the force normalized by the cell volume
172  fvSource[cellI] += sourceVec;
173  thrustSourceSum += fAxial * mesh_.V()[cellI];
174  torqueSourceSum += fCirc * mesh_.V()[cellI];
175  }
176 
177  reduce(thrustSourceSum, sumOp<scalar>());
178  reduce(torqueSourceSum, sumOp<scalar>());
179 
180  if (daOption_.getOption<word>("runStatus") == "solvePrimal")
181  {
182  if (mesh_.time().timeIndex() % printInterval_ == 0 || mesh_.time().timeIndex() == 1)
183  {
184  Info << "ThrustCoeff Source Term for " << diskName << ": " << thrustSourceSum << endl;
185  Info << "TorqueCoeff Source Term for " << diskName << ": " << torqueSourceSum << endl;
186  }
187  }
188  }
189 }

191 void DAFvSourceActuatorDisk::calcFvSourceCellIndices(HashTable<labelList>& fvSourceCellIndices)
192 {
193  /*
194  Description:
195  Calculate the lists of cell indices that are within the actuator disk space
196  NOTE: we support multiple actuator disks.
197 
198  Output:
199  fvSourceCellIndices: Hash table that contains the lists of cell indices that
200  are within the actuator disk space. The hash key is the name of the disk. We support
201  multiple disks. An example of fvSourceCellIndices can be:
202 
203  fvSourceCellIndices=
204  {
205  "disk1": {1,2,3,4,5},
206  "disk2": {6,7,8,9,10,11,12}
207  }
208 
209  Example:
210  An example of the fvSource in pyOptions in pyDAFoam can be
211  defOptions =
212  {
213  "fvSource"
214  {
215  "disk1"
216  {
217  "type": "actuatorDisk",
218  "source": cylinderAnnulusToCell,
219  "p1": [0.5, 0.3, 0.1], # p1 and p2 define the axis and width
220  "p2": [0.5, 0.3, 0.5],
221  "innerRadius": 0.1,
222  "outerRadius": 0.8,
223  "rotDir": "left",
224  "scale": 1.0,
225  "POD": 0.7
226  },
227  "disk2"
228  {
229  "type": "actuatorDisk",
230  "source": cylinderAnnulusToCell,
231  "p1": [0.0, 0.0, 0.1],
232  "p2": [0.0, 0.0, 0.5],
233  "innerRadius": 0.1,
234  "outerRadius": 0.8,
235  "rotDir": "right",
236  "scale": 0.1,
237  "POD": 1.0
238  }
239  }
240  }
241  */
242 
243  if (fvSourceSubDict_.toc().size() == 0)
244  {
245  FatalErrorIn("calcSourceCells") << "actuatorDisk is selected as fvSource "
246  << " but the options are empty!"
247  << abort(FatalError);
248  }
249 
250  forAll(fvSourceSubDict_.toc(), idxI)
251  {
252  word diskName = fvSourceSubDict_.toc()[idxI];
253 
254  fvSourceCellIndices.set(diskName, {});
255 
256  dictionary diskSubDict = fvSourceSubDict_.subDict(diskName);
257  word sourceType = diskSubDict.getWord("source");
258  // all avaiable source type are in src/meshTools/sets/cellSources
259  // Example of IO parameters os in applications/utilities/mesh/manipulation/topoSet
260  if (sourceType == "cylinderAnnulusToCell")
261  {
262  // create a topoSet
263  autoPtr<topoSet> currentSet(
264  topoSet::New(
265  "cellSet",
266  mesh_,
267  "set0",
268  IOobject::NO_READ));
269  // we need to change the min and max because they need to
270  // be of type point; however, we can't parse point type
271  // in pyDict, we need to change them here.
272 
273  scalarList point1;
274  scalarList point2;
275  diskSubDict.readEntry<scalarList>("p1", point1);
276  diskSubDict.readEntry<scalarList>("p2", point2);
277 
278  point p1;
279  point p2;
280  p1[0] = point1[0];
281  p1[1] = point1[1];
282  p1[2] = point1[2];
283  p2[0] = point2[0];
284  p2[1] = point2[1];
285  p2[2] = point2[2];
286 
287  scalar outerRadius = diskSubDict.getScalar("outerRadius");
288  scalar innerRadius = diskSubDict.getScalar("innerRadius");
289 
290  dictionary tmpDict;
291  tmpDict.set("p1", p1);
292  tmpDict.set("p2", p2);
293  tmpDict.set("innerRadius", innerRadius);
294  tmpDict.set("outerRadius", outerRadius);
295 
296  // create the source
297  autoPtr<topoSetSource> sourceSet(
298  topoSetSource::New(sourceType, mesh_, tmpDict));
299 
300  // add the sourceSet to topoSet
301  sourceSet().applyToSet(topoSetSource::NEW, currentSet());
302  // get the face index from currentSet, we need to use
303  // this special for loop
304  for (const label i : currentSet())
305  {
306  fvSourceCellIndices[diskName].append(i);
307  }
308  }
309  else
310  {
311  FatalErrorIn("calcFvSourceCells") << "source: " << sourceType << " not supported!"
312  << "Options are: cylinderAnnulusToCell!"
313  << abort(FatalError);
314  }
315  }
316 
317  if (daOption_.getOption<label>("debug"))
318  {
319  Info << "fvSourceCellIndices " << fvSourceCellIndices << endl;
320  }
321 }
