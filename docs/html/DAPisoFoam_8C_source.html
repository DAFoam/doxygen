76  stateBounaryAllInstances_.setSize(nTimeInstances_);

103 label DAPisoFoam::solvePrimal(
104  const Vec xvVec,
105  Vec wVec)
106 {
107  /*
108  Description:
109  Call the primal solver to get converged state variables
110 
111  Input:
112  xvVec: a vector that contains all volume mesh coordinates
113 
114  Output:
115  wVec: state variable vector
116  */
117 
118 #include "createRefsPiso.H"
119 
120  // change the run status
121  daOptionPtr_->setOption<word>("runStatus", "solvePrimal");
122 
123  // first check if we need to change the boundary conditions based on
124  // the primalBC dict in DAOption. NOTE: this will overwrite whatever
125  // boundary conditions defined in the "0" folder
126  dictionary bcDict = daOptionPtr_->getAllOptions().subDict("primalBC");
127  if (bcDict.toc().size() != 0)
128  {
129  Info << "Setting up primal boundary conditions based on pyOptions: " << endl;
130  daFieldPtr_->setPrimalBoundaryConditions();
131  }
132 
133  // call correctNut, this is equivalent to turbulence->validate();
134  daTurbulenceModelPtr_->updateIntermediateVariables();
135 
136  Info << "\nStarting time loop\n"
137  << endl;
138 
139  // deform the mesh based on the xvVec
140  this->pointVec2OFMesh(xvVec);
141 
142  // check mesh quality
143  label meshOK = this->checkMesh();
144 
145  if (!meshOK)
146  {
147  return 1;
148  }
149 
150  // We need to set the mesh moving to false, otherwise we will get V0 not found error.
151  // Need to dig into this issue later
152  mesh.moving(false);
153 
154  // create a file to store the objective values
155  this->initializeObjFuncHistFilePtr("objFuncHist.txt");
156 
157  primalMinRes_ = 1e10;
158  label printInterval = daOptionPtr_->getOption<label>("printIntervalUnsteady");
159  label printToScreen = 0;
160  label timeInstanceI = 0;
161  while (this->loop(runTime)) // using simple.loop() will have seg fault in parallel
162  {
163  printToScreen = this->isPrintTime(runTime, printInterval);
164 
165  if (printToScreen)
166  {
167  Info << "Time = " << runTime.timeName() << nl << endl;
168  }
169 
170  // Pressure-velocity PISO corrector
171  {
172 #include "UEqnPiso.H"
173 
174  // --- PISO loop
175  while (piso.correct())
176  {
177 #include "pEqnPiso.H"
178  }
179  }
180 
181  laminarTransport.correct();
182  daTurbulenceModelPtr_->correct();
183 
184  if (printToScreen)
185  {
186 #include "CourantNo.H"
187 
188  daTurbulenceModelPtr_->printYPlus();
189 
190  this->printAllObjFuncs();
191 
192  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
193  << " ClockTime = " << runTime.elapsedClockTime() << " s"
194  << nl << endl;
195  }
196 
197  // we write the objective function to file at every step
198  this->writeObjFuncHistFile();
199 
200  runTime.write();
201 
202  this->saveTimeInstanceField(timeInstanceI);
203 
204  }
205 
206  this->calcPrimalResidualStatistics("print");
207 
208  // primal converged, assign the OpenFoam fields to the state vec wVec
209  this->ofField2StateVec(wVec);
210 
211  // write the mesh to files
212  mesh.write();
213 
214  Info << "End\n"
215  << endl;
216 
217  return 0;
218 }

220 void DAPisoFoam::saveTimeInstanceField(label& timeInstanceI)
221 {
222  /*
223  Description:
224  Save primal variable to time instance list for unsteady adjoint
225  Here we save the last nTimeInstances snapshots
226  */
227 
228  scalar endTime = runTimePtr_->endTime().value();
229  scalar t = runTimePtr_->timeOutputValue();
230  scalar instanceStart =
231  endTime - periodicity_ / nTimeInstances_ * (nTimeInstances_ - 1 - timeInstanceI);
232 
233  // the 2nd condition is for t=9.999999999999 scenario)
234  if (t > instanceStart || fabs(t - endTime) < 1e-8)
235  {
236  Info << "Saving time instance " << timeInstanceI << " at Time = " << t << endl;
237 
238  // save fields
239  daFieldPtr_->ofField2List(
240  stateAllInstances_[timeInstanceI],
241  stateBounaryAllInstances_[timeInstanceI]);
242 
243  // save objective functions
244  forAll(daOptionPtr_->getAllOptions().subDict("objFunc").toc(), idxI)
245  {
246  word objFuncName = daOptionPtr_->getAllOptions().subDict("objFunc").toc()[idxI];
247  scalar objFuncVal = this->getObjFuncValue(objFuncName);
248  objFuncsAllInstances_[timeInstanceI].set(objFuncName, objFuncVal);
249  }
250 
251  // save runTime
252  runTimeAllInstances_[timeInstanceI] = t;
253  runTimeIndexAllInstances_[timeInstanceI] = runTimePtr_->timeIndex();
254 
255  if (daOptionPtr_->getOption<label>("debug"))
256  {
257  this->calcPrimalResidualStatistics("print");
258  }
259 
260  timeInstanceI++;
261  }
262  return;
263 }

265 void DAPisoFoam::setTimeInstanceField(const label instanceI)
266 {
267  /*
268  Description:
269  Assign primal variables based on the current time instance
270  If unsteady adjoint solvers are used, this virtual function should be
271  implemented in a child class, otherwise, return error if called
272  */
273 
274  Info << "Setting fields for time instance " << instanceI << endl;
275 
276  // set fields
277  daFieldPtr_->list2OFField(
278  stateAllInstances_[instanceI],
279  stateBounaryAllInstances_[instanceI]);
280 
281  // set run time
282  runTimePtr_->setTime(runTimeAllInstances_[instanceI], runTimeIndexAllInstances_[instanceI]);
283 
284  // We need to call correctBC multiple times to reproduce
285  // the exact residual for mulitpoint, this is needed for some boundary conditions
286  // and intermediate variables (e.g., U for inletOutlet, nut with wall functions)
287  for (label i = 0; i < 10; i++)
288  {
289  daResidualPtr_->correctBoundaryConditions();
290  daResidualPtr_->updateIntermediateVariables();
291  daModelPtr_->correctBoundaryConditions();
292  daModelPtr_->updateIntermediateVariables();
293  }
