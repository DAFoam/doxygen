85 void DAIntmdVar::update()
86 {
87  /*
88  Description:
89  Update the values for all intermediate variables defined in DAOption-intmdVar
90 
91  */
92 
93  if (hasIntmdVar_ != 0)
94  {
95  const dictionary intmdVarSubDict = daOption_.getAllOptions().subDict("intmdVar");
96 
97  forAll(intmdVarSubDict.toc(), idxI)
98  {
99  word varName = intmdVarSubDict.toc()[idxI];
100  dictionary varSubDict = intmdVarSubDict.subDict(varName);
101  word operation = varSubDict.get<word>("operation");
102 
103  if (operation == "Mean")
104  {
105  // Modified based on
106  // src/functionObjects/field/fieldAverage/fieldAverageItem/fieldAverageItemTemplates.C
107  word fieldType = varSubDict.get<word>("fieldType");
108  word baseField = varSubDict.get<word>("baseField");
109  label restartSteps = varSubDict.get<label>("restartSteps");
110  label nIdx = mesh_.time().timeIndex() % restartSteps;
111  if (fieldType == "volScalarField")
112  {
113  const volScalarField& baseState = mesh_.thisDb().lookupObject<volScalarField>(baseField);
114  volScalarField& intmdField = volScalarList_[idxI];
115  if (nIdx != 0)
116  {
117  // no need to compute Mean for nIdx = 0 because it will be assigned baseState
118  // when restart begines, i.e., nIdx = 1
119  scalar beta = 1.0 / nIdx;
120  intmdField = beta * baseState + (1.0 - beta) * intmdField;
121  }
122  }
123  else if (fieldType == "volVectorField")
124  {
125  const volVectorField& baseState = mesh_.thisDb().lookupObject<volVectorField>(baseField);
126  volVectorField& intmdField = volVectorList_[idxI];
127  if (nIdx != 0)
128  {
129  // no need to compute Mean for nIdx = 0 because it will be assigned baseState
130  // when restart begines, i.e., nIdx = 1
131  scalar beta = 1.0 / nIdx;
132  intmdField = beta * baseState + (1.0 - beta) * intmdField;
133  }
134  }
135  }
136  }
137  }
138 }
