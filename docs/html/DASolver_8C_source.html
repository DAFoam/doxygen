284 void DASolver::reduceStateResConLevel(
285  const dictionary& maxResConLv4JacPCMat,
286  HashTable<List<List<word>>>& stateResConInfo) const
287 {
288  /*
289  Description:
290  Reduce the connectivity levels for stateResConInfo
291  based on maxResConLv4JacPCMat specified in DAOption
292 
293  Input:
294  maxResConLv4JacPCMat: the maximal levels of connectivity for each
295  state variable residual
296 
297  Output:
298  stateResConInfo: reduced connectivity level.
299 
300  Example:
301 
302  If the original stateResConInfo reads:
303 
304  stateResConInfo
305  {
306  "URes":
307  {
308  {"U", "p", "phi"}, // level 0
309  {"U", "p"}, // level 1
310  {"U"} // level 2
311  }
312  }
313  And maxResConLv4JacPCMat in DAOption reads:
314 
315  maxResConLv4JacPCMat
316  {
317  "URes": 1
318  }
319 
320  Then, calling reduceStateResConLevel will give:
321 
322  stateResConInfo
323  {
324  "URes":
325  {
326  {"U", "p", "phi"}, // level 0
327  {"U", "p"}, // level 1
328  }
329  }
330 
331  Note that the level 2 of the connectivity in URes is removed becasue
332  "URes"=1 in maxResConLv4JacPCMat
333 
334  */
335 
336  // if no maxResConLv4JacPCMat is specified, just return;
337  if (maxResConLv4JacPCMat.toc().size() == 0)
338  {
339  Info << "maxResConLv4JacPCMat is empty, just return" << endl;
340  return;
341  }
342 
343  // now check if maxResConLv4JacPCMat has all the maxRes level defined
344  // and these max levels are <= stateResConInfo.size()
345  forAll(stateResConInfo.toc(), idxJ)
346  {
347  word key1 = stateResConInfo.toc()[idxJ];
348  bool keyFound = false;
349  forAll(maxResConLv4JacPCMat.toc(), idxI)
350  {
351  word key = maxResConLv4JacPCMat.toc()[idxI];
352  if (key == key1)
353  {
354  keyFound = true;
355  label maxLv = maxResConLv4JacPCMat.getLabel(key);
356  label maxLv1 = stateResConInfo[key1].size() - 1;
357  if (maxLv > maxLv1)
358  {
359  FatalErrorIn("") << "maxResConLv4JacPCMat maxLevel"
360  << maxLv << " for " << key
361  << " larger than stateResConInfo maxLevel "
362  << maxLv1 << " for " << key1
363  << abort(FatalError);
364  }
365  }
366  }
367  if (!keyFound)
368  {
369  FatalErrorIn("") << key1 << " not found in maxResConLv4JacPCMat"
370  << abort(FatalError);
371  }
372  }
373 
374  if (daOptionPtr_->getOption<label>("debug"))
375  {
376  Info << "Reducing max connectivity level of Jacobian PC Mat to : ";
377  Info << maxResConLv4JacPCMat << endl;
378  }
379 
380  // assign stateResConInfo to stateResConInfoBK
381  HashTable<List<List<word>>> stateResConInfoBK;
382  forAll(stateResConInfo.toc(), idxI)
383  {
384  word key = stateResConInfo.toc()[idxI];
385  stateResConInfoBK.set(key, stateResConInfo[key]);
386  }
387 
388  // now we can erase stateResConInfo
389  stateResConInfo.clearStorage();
390 
391  // get the reduced stateResConInfo
392  forAll(stateResConInfoBK.toc(), idxI)
393  {
394  word key = stateResConInfoBK.toc()[idxI];
395  label maxConLevel = maxResConLv4JacPCMat.getLabel(key);
396  label conSize = stateResConInfoBK[key].size();
397  if (conSize > maxConLevel + 1)
398  {
399  List<List<word>> conList;
400  conList.setSize(maxConLevel + 1);
401  for (label i = 0; i <= maxConLevel; i++) // NOTE: it is <=
402  {
403  conList[i] = stateResConInfoBK[key][i];
404  }
405  stateResConInfo.set(key, conList);
406  }
407  else
408  {
409  stateResConInfo.set(key, stateResConInfoBK[key]);
410  }
411  }
412  //Info<<stateResConInfo<<endl;
413 }

1834 void DASolver::setTotalDerivDict(
1835  const word objFuncName,
1836  const word designVarName,
1837  const Vec totalDerivVec,
1838  dictionary& totalDerivDict)
1839 {
1840  /*
1841  Description:
1842  Assign the totalDerivVec to totalDerivDict for a given objFuncName and
1843  designVariableName, such that we can get the total derivatives from
1844  the pyDAFoam.py for optimization
1845 
1846  Input:
1847  objFuncName: the name of the objective function
1848 
1849  designVarName: the name of the design variable
1850 
1851  totalDerivVec: the total derivative vector obtained from DASolver::calcTotalDeriv
1852 
1853  Output:
1854  totalDerivDict: the dictionary to store the total derivative values.
1855  Note: this list store derivative wrt all the design variables and is reduced
1856  across all processors. This is necessary because the pyDAFoam.py will get the total
1857  derivative from totalDerivDict_ and ask for totalDeriv for all design variables,
1858  not just for the design variable stored in the local totalDerivVec. We need
1859  to scatter the totalDerivVec and gather all the values
1860 
1861  */
1862 
1863  label vecSize;
1864  VecGetSize(totalDerivVec, &vecSize);
1865 
1866  // scatter colors to local array for all procs
1867  Vec vout;
1868  VecScatter ctx;
1869  VecScatterCreateToAll(totalDerivVec, &ctx, &vout);
1870  VecScatterBegin(ctx, totalDerivVec, vout, INSERT_VALUES, SCATTER_FORWARD);
1871  VecScatterEnd(ctx, totalDerivVec, vout, INSERT_VALUES, SCATTER_FORWARD);
1872 
1873  PetscScalar* voutArray;
1874  VecGetArray(vout, &voutArray);
1875 
1876  scalarList totalDerivList;
1877 
1878  for (label i = 0; i < vecSize; i++)
1879  {
1880  totalDerivList.append(voutArray[i]);
