98 label DASolver::loop(Time& runTime)

101  Description:
102  The loop method to increment the runtime. The reason we implent this is
103  because the runTime.loop() and simple.loop() give us seg fault...
104  */
105 
106  scalar endTime = runTime.endTime().value();
107  scalar deltaT = runTime.deltaT().value();
108  scalar t = runTime.timeOutputValue();
109  scalar tol = daOptionPtr_->getOption<scalar>("primalMinResTol");
110 
111  // execute functionObjectList, e.g., field averaging, sampling
112  functionObjectList& funcObj = const_cast<functionObjectList&>(runTime.functionObjects());
113  if (runTime.timeIndex() == runTime.startTimeIndex())
114  {
115  funcObj.start();
116  }
117  else
118  {
119  funcObj.execute();
120  }
121 
122  // check exit condition
123  if (primalMinRes_ < tol)
124  {
125  Info << "Time = " << t << endl;
126  Info << "Minimal residual " << primalMinRes_ << " satisfied the prescribed tolerance " << tol << endl
127  << endl;
128  this->printAllObjFuncs();
129  runTime.writeNow();
130  prevPrimalSolTime_ = t;
131  funcObj.end();
132  return 0;
133  }
134  else if (t > endTime - 0.5 * deltaT)
135  {
136  prevPrimalSolTime_ = t;
137  funcObj.end();
138  return 0;
139  }
140  else

319 void DASolver::reduceStateResConLevel(
320  const dictionary& maxResConLv4JacPCMat,
321  HashTable<List<List<word>>>& stateResConInfo) const
322 {
323  /*
324  Description:
325  Reduce the connectivity levels for stateResConInfo
326  based on maxResConLv4JacPCMat specified in DAOption
327 
328  Input:
329  maxResConLv4JacPCMat: the maximal levels of connectivity for each
330  state variable residual
331 
332  Output:
333  stateResConInfo: reduced connectivity level.
334 
335  Example:
336 
337  If the original stateResConInfo reads:
338 
339  stateResConInfo
340  {
341  "URes":
342  {
343  {"U", "p", "phi"}, // level 0
344  {"U", "p"}, // level 1
345  {"U"} // level 2
346  }
347  }
348  And maxResConLv4JacPCMat in DAOption reads:
349 
350  maxResConLv4JacPCMat
351  {
352  "URes": 1
353  }
354 
355  Then, calling reduceStateResConLevel will give:
356 
357  stateResConInfo
358  {
359  "URes":
360  {
361  {"U", "p", "phi"}, // level 0
362  {"U", "p"}, // level 1
363  }
364  }
365 
366  Note that the level 2 of the connectivity in URes is removed becasue
367  "URes"=1 in maxResConLv4JacPCMat
368 
369  */
370 
371  // if no maxResConLv4JacPCMat is specified, just return;
372  if (maxResConLv4JacPCMat.toc().size() == 0)
373  {
374  Info << "maxResConLv4JacPCMat is empty, just return" << endl;
375  return;
376  }
377 
378  // now check if maxResConLv4JacPCMat has all the maxRes level defined
379  // and these max levels are <= stateResConInfo.size()
380  forAll(stateResConInfo.toc(), idxJ)
381  {
382  word key1 = stateResConInfo.toc()[idxJ];
383  bool keyFound = false;
384  forAll(maxResConLv4JacPCMat.toc(), idxI)
385  {
386  word key = maxResConLv4JacPCMat.toc()[idxI];
387  if (key == key1)
388  {
389  keyFound = true;
390  label maxLv = maxResConLv4JacPCMat.getLabel(key);
391  label maxLv1 = stateResConInfo[key1].size() - 1;
392  if (maxLv > maxLv1)
393  {
394  FatalErrorIn("") << "maxResConLv4JacPCMat maxLevel"
395  << maxLv << " for " << key
396  << " larger than stateResConInfo maxLevel "
397  << maxLv1 << " for " << key1
398  << abort(FatalError);
399  }
400  }
401  }
402  if (!keyFound)
403  {
404  FatalErrorIn("") << key1 << " not found in maxResConLv4JacPCMat"
405  << abort(FatalError);
406  }
407  }
408 
409  if (daOptionPtr_->getOption<label>("debug"))
410  {
411  Info << "Reducing max connectivity level of Jacobian PC Mat to : ";
412  Info << maxResConLv4JacPCMat << endl;
413  }
414 
415  // assign stateResConInfo to stateResConInfoBK
416  HashTable<List<List<word>>> stateResConInfoBK;
417  forAll(stateResConInfo.toc(), idxI)
418  {
419  word key = stateResConInfo.toc()[idxI];
420  stateResConInfoBK.set(key, stateResConInfo[key]);
421  }
422 
423  // now we can erase stateResConInfo
424  stateResConInfo.clearStorage();
425 
426  // get the reduced stateResConInfo
427  forAll(stateResConInfoBK.toc(), idxI)
428  {
429  word key = stateResConInfoBK.toc()[idxI];
430  label maxConLevel = maxResConLv4JacPCMat.getLabel(key);
431  label conSize = stateResConInfoBK[key].size();
432  if (conSize > maxConLevel + 1)
433  {
434  List<List<word>> conList;
435  conList.setSize(maxConLevel + 1);
436  for (label i = 0; i <= maxConLevel; i++) // NOTE: it is <=
437  {
438  conList[i] = stateResConInfoBK[key][i];
439  }
440  stateResConInfo.set(key, conList);
441  }
442  else
443  {
444  stateResConInfo.set(key, stateResConInfoBK[key]);
445  }
446  }
447  //Info<<stateResConInfo<<endl;
448 }
